## Why It Surfaces

This pattern appears whenever the query range is defined over a sortable key such as price, timestamp, or coordinate. After sorting by that key, every query becomes an index interval, so the remaining problem is a standard range maximum query.

## Worked Example

For items `[(24, 6), (30, 8), (21, 7)]`, sort by price into `[(21, 7), (24, 6), (30, 8)]`. The price query `[10, 24]` covers the first two items, so the maximum weight is `7`. The query `[20, 30]` covers all three, so the answer is `8`.

  let left = 0;
  let right = values.length;

  while (left < right) {
    const mid = Math.floor((left + right) / 2);

    } else {
      right = mid;

class MaxWeightByPriceRange {
  constructor(items) {
    if (!Array.isArray(items) || items.length === 0) {
      throw new RangeError('Items must be a non-empty array');
    }

    this.items = items
      .map(([price, weight]) => ({ price, weight }))
      .sort((left, right) => left.price - right.price);
    this.prices = this.items.map((item) => item.price);
    this.weights = this.items.map((item) => item.weight);
    this.size = this.items.length;
    this.tree = new Array(this.size * 4).fill(Number.NEGATIVE_INFINITY);
    this.build(1, 0, this.size - 1);
  }

  build(node, left, right) {
    if (left === right) {
      this.tree[node] = this.weights[left];
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = Math.max(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  query(minPrice, maxPrice) {
    if (minPrice > maxPrice) {
      throw new RangeError('Price range is invalid');
    }

    const left = lowerBound(this.prices, minPrice);
    const right = upperBound(this.prices, maxPrice) - 1;

    if (left > right) {
      return null;
    }

    return this.queryNode(1, 0, this.size - 1, left, right);
  }

  queryNode(node, left, right, queryLeft, queryRight) {
    if (queryRight < left || right < queryLeft) {
      return Number.NEGATIVE_INFINITY;
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return Math.max(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }
}

class MaxWeightByPriceRange {
  private readonly prices: number[];
  private readonly weights: number[];
  private readonly size: number;
  private readonly tree: number[];

  constructor(items: Array<[number, number]>) {
    if (items.length === 0) {
      throw new RangeError('Items must be a non-empty array');
    }

    const sorted = items
      .map(([price, weight]) => ({ price, weight }))
      .sort((left, right) => left.price - right.price);

    this.prices = sorted.map((item) => item.price);
    this.weights = sorted.map((item) => item.weight);
    this.size = sorted.length;
    this.tree = new Array<number>(this.size * 4).fill(Number.NEGATIVE_INFINITY);
    this.build(1, 0, this.size - 1);
  }

  query(minPrice: number, maxPrice: number): number | null {
    if (minPrice > maxPrice) {
      throw new RangeError('Price range is invalid');
    }

    const left = lowerBound(this.prices, minPrice);
    const right = upperBound(this.prices, maxPrice) - 1;

    if (left > right) {
      return null;
    }

    return this.queryNode(1, 0, this.size - 1, left, right);
  }

  private build(node: number, left: number, right: number): void {
    if (left === right) {
      this.tree[node] = this.weights[left];
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = Math.max(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  private queryNode(
    node: number,
    left: number,
    right: number,
    queryLeft: number,
    queryRight: number,
  ): number {
    if (queryRight < left || right < queryLeft) {
      return Number.NEGATIVE_INFINITY;
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return Math.max(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }
}

export function findElementWithMaximumWeightInGivenPriceRangeForQQueries(
  items: Array<[number, number]>,
  queries: Array<[number, number]>,
): Array<number | null> {
  const index = new MaxWeightByPriceRange(items);
  return queries.map(([minPrice, maxPrice]) => index.query(minPrice, maxPrice));
}

## Common Pitfalls

- Forgetting to sort by price before building the range structure.
- Using one binary search bound for both sides of the interval.
- Ignoring the case where no item falls inside the requested price range.
