## Why It Surfaces

This is a clean simulation problem with fixed rules and one evolving state variable. The useful reduction is to track only the weekday of the first day of each month.

## Worked Example

If `1 Jan 1901` is a Tuesday, then the weekday for the first of the next month is just `(currentWeekday + daysInMonth) mod 7`. Count each month whose starting weekday is Sunday.

function isLeapYear(year) {
  return (year % 4 === 0 && year % 100 !== 0) || year % 400 === 0;
}

function daysInMonth(year, month) {
  if (month === 2) {
    return isLeapYear(year) ? 29 : 28;
  }

  if ([4, 6, 9, 11].includes(month)) {
    return 30;
  }

  return 31;
}

function problem019(startYear = 1901, endYear = 2000) {
  if (!Number.isInteger(startYear) || !Number.isInteger(endYear) || startYear > endYear) {
    throw new RangeError('Year range must be valid integers with startYear <= endYear');
  }

  let firstDayOfMonth = 2;
  let sundays = 0;

  for (let year = 1901; year < startYear; year += 1) {
    for (let month = 1; month <= 12; month += 1) {
      firstDayOfMonth = (firstDayOfMonth + daysInMonth(year, month)) % 7;
    }
  }

  for (let year = startYear; year <= endYear; year += 1) {
    for (let month = 1; month <= 12; month += 1) {
      if (firstDayOfMonth === 0) {
        sundays += 1;
      }

      firstDayOfMonth = (firstDayOfMonth + daysInMonth(year, month)) % 7;
    }
  }

  return sundays;
}

## Follow-up Questions

## Common Pitfalls

- Misapplying the century leap-year rule.
- Forgetting that the count is for first-of-month Sundays only.
- Starting from the wrong weekday anchor for `1 Jan 1901`.
