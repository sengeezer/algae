## Why It Surfaces

Karatsuba is one of the cleanest examples of an algebraic trick changing a recurrence. The interview value is understanding why three recursive products are enough, not memorizing another multiplication routine.

## Worked Example

The binary strings `1100` and `1010` represent `12` and `10`, so their product is `120`. The function returns `120n` as an exact `BigInt` product.

function validateBinaryOperand(value, name) {
  if (typeof value !== 'string' || !/^[01]+$/.test(value)) {
    throw new TypeError(`${name} must be a non-empty binary string`);

}

function bitLength(value) {
  let number = value;
  let length = 0;

  while (number > 0n) {
    number >>= 1n;
    length += 1;
  }

  return length || 1;
}

function karatsubaMultiply(left, right) {
  if (left === 0n || right === 0n) {
    return 0n;
  }

  if (left < 2n || right < 2n) {
    return left * right;
  }

  const length = Math.max(bitLength(left), bitLength(right));
  const half = BigInt(Math.ceil(length / 2));
  const mask = (1n << half) - 1n;
  const highLeft = left >> half;
  const lowLeft = left & mask;
  const highRight = right >> half;
  const lowRight = right & mask;

  const z0 = karatsubaMultiply(lowLeft, lowRight);
  const z2 = karatsubaMultiply(highLeft, highRight);
  const z1 = karatsubaMultiply(lowLeft + highLeft, lowRight + highRight) - z2 - z0;

  return (z2 << (half * 2n)) + (z1 << half) + z0;
}

function karatsubaAlgorithmForFastMultiplicationUsingDivideAndConquerAlgorithm(left, right) {

  validateBinaryOperand(right, 'right');

  const leftValue = BigInt(`0b${left}`);
  const rightValue = BigInt(`0b${right}`);

}

  if (left === 0n || right === 0n) {
    return 0n;
  }

  if (left < 2n || right < 2n) {
    return left * right;

  const highRight = right >> half;
  const lowRight = right & mask;

  const z0 = karatsubaMultiply(lowLeft, lowRight);
  const z2 = karatsubaMultiply(highLeft, highRight);

}

export function karatsubaAlgorithmForFastMultiplicationUsingDivideAndConquerAlgorithm(

## Follow-up Questions

- Return the product as a binary string instead of a `BigInt`.
- Compare Karatsuba with schoolbook multiplication and FFT-based multiplication.
- Explain why the middle term only needs one additional recursive product.

## Common Pitfalls

- Recursing into four half-size products and losing the whole point of Karatsuba.
- Mishandling odd bit lengths during splitting.
- Using built-in big-number multiplication for the full recursive step instead of only the base case.
