## Why It Surfaces

Affine cipher questions are really modular-arithmetic questions in disguise. The interesting part is not the alphabet lookup itself, but the fact that decryption only works when the multiplicative coefficient has an inverse modulo 26.

## Worked Example

If `a = 5` and `b = 8`, the letter `A` maps to index `0`, so the encrypted index is `(5 * 0 + 8) mod 26 = 8`, which becomes `I`.

```js
function affineCipher(value, a, b, mode = 'encrypt') {
  if (typeof value !== 'string') {
    throw new TypeError('Value must be a string');
  }

  if (!Number.isInteger(a) || !Number.isInteger(b)) {
    throw new TypeError('Coefficients must be integers');
  }

  function gcd(left, right) {
    while (right !== 0) {
      [left, right] = [right, left % right];
    }

    return Math.abs(left);
  }

  function mod(n, m) {
    return ((n % m) + m) % m;
  }

  function modInverse(number, modulus) {
    for (let candidate = 1; candidate < modulus; candidate += 1) {
      if (mod(number * candidate, modulus) === 1) {
        return candidate;
      }
    }

    throw new Error('Coefficient a must be coprime with 26');
  }

  if (gcd(a, 26) !== 1) {
    throw new Error('Coefficient a must be coprime with 26');
  }

  const inverse = modInverse(a, 26);

  return value.replace(/[a-z]/gi, (char) => {
    const base = /[A-Z]/.test(char) ? 65 : 97;
    const index = char.charCodeAt(0) - base;
    const nextIndex = mode === 'encrypt'
      ? mod(a * index + b, 26)
      : mod(inverse * (index - b), 26);

    return String.fromCharCode(base + nextIndex);
  });
}
```

export function affineCipher(
  value: string,
  a: number,
  b: number,
  mode: 'encrypt' | 'decrypt' = 'encrypt',
): string {
  function gcd(left: number, right: number): number {
    while (right !== 0) {
      [left, right] = [right, left % right];
    }

    return Math.abs(left);
  }

  function mod(n: number, m: number): number {
    return ((n % m) + m) % m;
  }

  function modInverse(number: number, modulus: number): number {
    for (let candidate = 1; candidate < modulus; candidate += 1) {
      if (mod(number * candidate, modulus) === 1) {
        return candidate;
      }
    }

    throw new Error('Coefficient a must be coprime with 26');
  }

  if (gcd(a, 26) !== 1) {
    throw new Error('Coefficient a must be coprime with 26');
  }

  const inverse = modInverse(a, 26);

  return value.replace(/[a-z]/gi, (char) => {
    const base = /[A-Z]/.test(char) ? 65 : 97;
    const index = char.charCodeAt(0) - base;
    const nextIndex = mode === 'encrypt'
      ? mod(a * index + b, 26)
      : mod(inverse * (index - b), 26);

    return String.fromCharCode(base + nextIndex);
  });
}

## Common Pitfalls

- Forgetting the coprime requirement for the multiplicative coefficient.
- Returning sentinel placeholders for spaces instead of preserving non-letters directly.
- Reversing the additive part without also applying the modular inverse during decryption.
