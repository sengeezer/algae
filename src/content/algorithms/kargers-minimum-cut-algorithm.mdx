## Why It Surfaces

Karger's algorithm stands out because it solves a cut problem by embracing randomness rather than avoiding it. Instead of carefully choosing which edge to keep or cut, it repeatedly contracts random edges and relies on probability plus repetition to preserve the true minimum cut often enough.

## Worked Example

Each contraction merges two endpoints into one supernode and removes any self-loops that arise. When only two supernodes remain, every edge still crossing between them belongs to that run's candidate cut; repeating the process many times drives up the chance of seeing the true global minimum cut.

function kargersMinimumCutAlgorithm(vertexCount, edges, random = Math.random) {
  class UnionFind {
    constructor(size) {
      this.parent = Array.from({ length: size }, (_, index) => index);
      this.rank = new Array(size).fill(0);
    }

    find(node) {
      if (this.parent[node] !== node) {
        this.parent[node] = this.find(this.parent[node]);
      }
      return this.parent[node];
    }

    union(left, right) {
      const rootLeft = this.find(left);
      const rootRight = this.find(right);
      if (rootLeft === rootRight) {
        return false;
      }

      if (this.rank[rootLeft] < this.rank[rootRight]) {
        this.parent[rootLeft] = rootRight;
      } else if (this.rank[rootLeft] > this.rank[rootRight]) {
        this.parent[rootRight] = rootLeft;
      } else {
        this.parent[rootRight] = rootLeft;
        this.rank[rootLeft] += 1;
      }

      return true;
    }
  }

  const dsu = new UnionFind(vertexCount);
  let components = vertexCount;

  while (components > 2) {
    const [from, to] = edges[Math.floor(random() * edges.length)];
    if (dsu.union(from, to)) {
      components -= 1;
    }
  }

  let cut = 0;
  for (const [from, to] of edges) {
    if (dsu.find(from) !== dsu.find(to)) {
      cut += 1;
    }
  }

  return cut;
}

class UnionFind {
  private readonly parent: number[];
  private readonly rank: number[];

  constructor(size: number) {
    this.parent = Array.from({ length: size }, (_, index) => index);
    this.rank = new Array<number>(size).fill(0);
  }

  find(node: number): number {
    if (this.parent[node] !== node) {
      this.parent[node] = this.find(this.parent[node]);
    }
    return this.parent[node];
  }

  union(left: number, right: number): boolean {
    const rootLeft = this.find(left);
    const rootRight = this.find(right);
    if (rootLeft === rootRight) {
      return false;
    }

    if (this.rank[rootLeft] < this.rank[rootRight]) {
      this.parent[rootLeft] = rootRight;
    } else if (this.rank[rootLeft] > this.rank[rootRight]) {
      this.parent[rootRight] = rootLeft;
    } else {
      this.parent[rootRight] = rootLeft;
      this.rank[rootLeft] += 1;
    }

    return true;
  }
}

export function kargersMinimumCutAlgorithm(
  vertexCount: number,
  edges: Array<[number, number]>,
  random: () => number = Math.random,
): number {
  const dsu = new UnionFind(vertexCount);
  let components = vertexCount;

  while (components > 2) {
    const [from, to] = edges[Math.floor(random() * edges.length)];
    if (dsu.union(from, to)) {
      components -= 1;
    }
  }

  let cut = 0;
  for (const [from, to] of edges) {
    if (dsu.find(from) !== dsu.find(to)) {
      cut += 1;
    }
  }

  return cut;
}

## Follow-up Questions

- Explain why one run is probabilistic and why repetition improves confidence.
- Compare global min cut from Karger with source-sink min cut from max flow.
- Return the actual partition of the surviving two supernodes, not just the cut size.

## Common Pitfalls

- Assuming one randomized run is guaranteed to find the true minimum cut.
- Forgetting to ignore self-loops after contractions.
- Using Karger for directed source-sink cut questions where the formulation is different.
