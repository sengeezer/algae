## Why It Surfaces

This is the practical TSP follow-up after the exact Held-Karp version. Instead of searching every subset state, you build an MST, walk it in preorder, and use triangle inequality to shortcut revisits without making the tour worse than twice the optimum.

## Worked Example

For a metric distance matrix on four cities, suppose Prim's algorithm builds MST edges `(0, 1)`, `(1, 2)`, and `(1, 3)`. A preorder walk from `0` visits cities in order `[0, 1, 2, 3]`, and closing the cycle gives the approximate tour `[0, 1, 2, 3, 0]`.

function approximateSolutionForTravellingSalesmanProblemUsingMST(distances) {
  const cityCount = distances.length;
  if (cityCount === 0) {
    return { cost: 0, tour: [] };
  }

  const key = new Array(cityCount).fill(Number.POSITIVE_INFINITY);
  const parent = new Array(cityCount).fill(-1);
  const inMst = new Array(cityCount).fill(false);
  key[0] = 0;

  for (let added = 0; added < cityCount; added += 1) {
    let nextCity = -1;

    for (let city = 0; city < cityCount; city += 1) {
      if (!inMst[city] && (nextCity === -1 || key[city] < key[nextCity])) {
        nextCity = city;
      }
    }

    inMst[nextCity] = true;

    for (let neighbor = 0; neighbor < cityCount; neighbor += 1) {
      const weight = distances[nextCity][neighbor];
      if (!inMst[neighbor] && neighbor !== nextCity && weight < key[neighbor]) {
        key[neighbor] = weight;
        parent[neighbor] = nextCity;
      }
    }
  }

  const tree = Array.from({ length: cityCount }, () => []);
  for (let city = 1; city < cityCount; city += 1) {
    tree[parent[city]].push(city);
  }

  const preorder = [];

  function dfs(city) {
    preorder.push(city);
    for (const neighbor of tree[city]) {
      dfs(neighbor);
    }
  }

  dfs(0);
  const tour = [...preorder, 0];
  let cost = 0;

  for (let index = 0; index + 1 < tour.length; index += 1) {
    cost += distances[tour[index]][tour[index + 1]];
  }

  return { cost, tour };
}

## Common Pitfalls

- Applying the guarantee to non-metric graphs where shortcutting can break the bound.
- Forgetting to return to the starting city after the preorder walk.
- Mixing this approximation with the exact TSP dynamic-programming solution.
