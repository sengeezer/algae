## Why It Surfaces

This is the kind of prompt that tries to hide a basic range query behind notation. Once you notice that `(a ^ b) | xnor(a, b)` sets every bit from `0` through the largest most-significant bit present in either value, the whole query collapses into: find the maximum value in `[L, R]`, then turn that value into an all-ones mask.

## Worked Example

For `arr = [5, 2, 3, 0]`, the query `[1, 3]` looks at adjacent pairs `(2, 3)` and `(3, 0)`. Both pairs produce masks whose OR is `3`. The query `[0, 2]` sees the maximum value `5`, whose mask is `7`, so the answer becomes `7`.

function fullMaskFromMax(value) {
  if (!Number.isSafeInteger(value) || value < 0) {
    throw new RangeError('Values must be safe non-negative integers');
  }

  if (value === 0) {
    return 0;
  }

  return 2 ** (Math.floor(Math.log2(value)) + 1) - 1;
}

class MaxSegmentTree {
  constructor(values) {
    if (!Array.isArray(values) || values.length === 0) {
      throw new RangeError('Values must be a non-empty array');
    }

    this.values = values.slice();
    this.size = values.length;
    this.tree = new Array(this.size * 4).fill(Number.NEGATIVE_INFINITY);

    for (const value of this.values) {
      if (!Number.isSafeInteger(value) || value < 0) {
        throw new RangeError('Values must be safe non-negative integers');
      }
    }

    this.build(1, 0, this.size - 1);
  }

  build(node, left, right) {
    if (left === right) {
      this.tree[node] = this.values[left];
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = Math.max(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  query(queryLeft, queryRight) {
    this.validateRange(queryLeft, queryRight);
    return this.queryNode(1, 0, this.size - 1, queryLeft, queryRight);
  }

  queryNode(node, left, right, queryLeft, queryRight) {
    if (queryRight < left || right < queryLeft) {
      return Number.NEGATIVE_INFINITY;
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return Math.max(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }

  validateRange(queryLeft, queryRight) {
    if (
      !Number.isInteger(queryLeft)
      || !Number.isInteger(queryRight)
      || queryLeft < 0
      || queryRight < queryLeft
      || queryRight >= this.size
    ) {
      throw new RangeError('Query range is invalid');
    }
  }
}

function queriesToEvaluateTheGivenEquation(values, queries) {
  const tree = new MaxSegmentTree(values);

  return queries.map(([left, right]) => {
    const maxValue = tree.query(left, right);
    return fullMaskFromMax(maxValue);
  });
}

  if (!Number.isSafeInteger(value) || value < 0) {
    throw new RangeError('Values must be safe non-negative integers');
  }

  if (value === 0) {
    return 0;
  }

  return 2 ** (Math.floor(Math.log2(value)) + 1) - 1;
}

class MaxSegmentTree {
  private readonly values: number[];
  private readonly size: number;
  private readonly tree: number[];

  constructor(values: number[]) {
    if (values.length === 0) {
      throw new RangeError('Values must be a non-empty array');
    }

    this.values = values.slice();
    this.size = values.length;
    this.tree = new Array<number>(this.size * 4).fill(Number.NEGATIVE_INFINITY);

    for (const value of this.values) {
      if (!Number.isSafeInteger(value) || value < 0) {
        throw new RangeError('Values must be safe non-negative integers');
      }
    }

    this.build(1, 0, this.size - 1);
  }

  query(queryLeft: number, queryRight: number): number {
    this.validateRange(queryLeft, queryRight);
    return this.queryNode(1, 0, this.size - 1, queryLeft, queryRight);
  }

  private build(node: number, left: number, right: number): void {
    if (left === right) {
      this.tree[node] = this.values[left];
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = Math.max(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  private queryNode(
    node: number,
    left: number,
    right: number,
    queryLeft: number,
    queryRight: number,
  ): number {
    if (queryRight < left || right < queryLeft) {
      return Number.NEGATIVE_INFINITY;
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return Math.max(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }

  private validateRange(queryLeft: number, queryRight: number): void {
    if (
      !Number.isInteger(queryLeft)
      || !Number.isInteger(queryRight)
      || queryLeft < 0
      || queryRight < queryLeft
      || queryRight >= this.size
    ) {
      throw new RangeError('Query range is invalid');
    }
  }
}

export function queriesToEvaluateTheGivenEquation(
  values: number[],
  queries: Array<[number, number]>,
): number[] {
  const tree = new MaxSegmentTree(values);
  return queries.map(([left, right]) => fullMaskFromMax(tree.query(left, right)));
}

## Common Pitfalls

- Treating the formula literally and doing too much per query.
- Forgetting that the answer depends on the maximum value in `[L, R]`, not `[L, R - 1]`.
- Using JavaScript bitwise operators directly on large values and running into 32-bit coercion.
