## Why It Surfaces

Every substring of the text is a prefix of some suffix. That means a trie built from all suffixes can answer repeated pattern queries by walking the pattern once and then returning the start positions cached at the final node.

## Worked Example

For text `banana`, inserting all suffixes causes the path for `ana` to collect start indices `1` and `3`. Searching `ana` walks `a -> n -> a` and returns `[1, 3]`.

```js
class SuffixTrieNode {
  constructor() {
    this.children = new Map();
    this.positions = [];
  }
}

function buildSuffixTrie(text) {
  const root = new SuffixTrieNode();

  for (let start = 0; start < text.length; start += 1) {
    let node = root;

    for (let index = start; index < text.length; index += 1) {
      const char = text[index];

      if (!node.children.has(char)) {
        node.children.set(char, new SuffixTrieNode());
      }

      node = node.children.get(char);
      node.positions.push(start);
    }
  }

  return root;
}

function patternSearchingUsingATrieOfAllSuffixes(text, pattern) {
  if (typeof text !== 'string' || typeof pattern !== 'string') {
    throw new TypeError('Text and pattern must be strings');
  }

  if (pattern.length === 0) {
    throw new RangeError('Pattern must be non-empty');
  }

  const root = buildSuffixTrie(text);
  let node = root;

  for (const char of pattern) {
    node = node.children.get(char);
    if (!node) {
      return [];
    }
  }

  return [...node.positions];
}
```

```ts
class SuffixTrieNode {
  children = new Map<string, SuffixTrieNode>();
  positions: number[] = [];
}

function buildSuffixTrie(text: string): SuffixTrieNode {
  const root = new SuffixTrieNode();

  for (let start = 0; start < text.length; start += 1) {
    let node = root;

    for (let index = start; index < text.length; index += 1) {
      const char = text[index];

      if (!node.children.has(char)) {
        node.children.set(char, new SuffixTrieNode());
      }

      node = node.children.get(char) as SuffixTrieNode;
      node.positions.push(start);
    }
  }

  return root;
}

export function patternSearchingUsingATrieOfAllSuffixes(text: string, pattern: string): number[] {
  if (pattern.length === 0) {
    throw new RangeError('Pattern must be non-empty');
  }

  const root = buildSuffixTrie(text);
  let node: SuffixTrieNode | undefined = root;

  for (const char of pattern) {
    node = node.children.get(char);
    if (!node) {
      return [];
    }
  }

  return [...node.positions];
}
```

## Follow-up Questions

- How would a suffix tree improve the O(n²) space usage?
- What if you need only the first occurrence instead of every match?
- How would you reuse the built suffix trie across many pattern searches?

## Common Pitfalls

- Storing only one position per node and losing repeated matches.
- Returning end indices when the caller expects start indices.
- Rebuilding the suffix trie per query in a system that should reuse preprocessing.
