## Why It Surfaces

BFS is the shortest-path tool for unweighted graphs and state spaces. If the prompt says minimum number of steps, nearest distance, or level-order traversal, BFS is often the first correct answer.

## JavaScript

```js
function bfs(graph, start) {
  const queue = [start];
  const visited = new Set([start]);
  const order = [];

  while (queue.length > 0) {
    const node = queue.shift();
    order.push(node);

    for (const neighbor of graph.get(node) ?? []) {
      if (visited.has(neighbor)) {
        continue;
      }

      visited.add(neighbor);
      queue.push(neighbor);
    }
  }

  return order;
}
```

## TypeScript

```ts
type Point = [row: number, column: number];

export function shortestPath(grid: number[][], start: Point, target: Point): number {
  const queue: Array<[Point, number]> = [[start, 0]];
  const visited = new Set([start.join(":")]);
  const directions: Point[] = [[1, 0], [-1, 0], [0, 1], [0, -1]];

  while (queue.length > 0) {
    const [[row, column], steps] = queue.shift()!;

    if (row === target[0] && column === target[1]) {
      return steps;
    }

    for (const [dr, dc] of directions) {
      const nextRow = row + dr;
      const nextColumn = column + dc;
      const key = `${nextRow}:${nextColumn}`;

      if (
        nextRow < 0 ||
        nextRow >= grid.length ||
        nextColumn < 0 ||
        nextColumn >= grid[0].length ||
        grid[nextRow][nextColumn] === 1 ||
        visited.has(key)
      ) {
        continue;
      }

      visited.add(key);
      queue.push([[nextRow, nextColumn], steps + 1]);
    }
  }

  return -1;
}
```

## Interview Notes

- Mark nodes visited when they are enqueued, not when they are popped.
- Mention multi-source BFS when several starting cells spread at once.
- Contrast BFS with Dijkstra as soon as weighted edges appear.
