## Why It Surfaces

This algorithm shows up as a practical engineering alternative once you already know Dijkstra. The core idea is to use a deque and a more aggressive revisit policy so nodes with newly improved distances can jump ahead of the ordinary processing order.

## Worked Example

Suppose vertex `b` was already seen once, but a relaxation through vertex `a` improves its distance again. Instead of waiting at the back like a fresh discovery, `b` moves to the front so its better distance can propagate quickly through the graph.

```js
function desopoPapeAlgorithmSingleSourceShortestPath(nodeCount, edges, source) {
  const graph = Array.from({ length: nodeCount }, () => []);
  for (const [from, to, weight] of edges) {
    graph[from].push([to, weight]);
    graph[to].push([from, weight]);
  }

  const distance = new Array(nodeCount).fill(Number.POSITIVE_INFINITY);
  const inQueue = new Array(nodeCount).fill(false);
  const wasPopped = new Array(nodeCount).fill(false);
  const deque = [source];
  distance[source] = 0;
  inQueue[source] = true;

  while (deque.length > 0) {
    const node = deque.shift();
    inQueue[node] = false;
    wasPopped[node] = true;

    for (const [neighbor, weight] of graph[node]) {
      const candidate = distance[node] + weight;
      if (candidate >= distance[neighbor]) {
        continue;
      }

      distance[neighbor] = candidate;
      if (inQueue[neighbor]) {
        continue;
      }

      if (wasPopped[neighbor]) {
        deque.unshift(neighbor);
      } else {
        deque.push(neighbor);
      }
      inQueue[neighbor] = true;
    }
  }

  return distance;
}
```

```ts
  nodeCount: number,
  edges: Array<[number, number, number]>,
  source: number,
): number[] {
  const graph = Array.from({ length: nodeCount }, () => [] as Array<[number, number]>);
  for (const [from, to, weight] of edges) {
    graph[from].push([to, weight]);
    graph[to].push([from, weight]);
  }

  const distance = new Array<number>(nodeCount).fill(Number.POSITIVE_INFINITY);
  const inQueue = new Array<boolean>(nodeCount).fill(false);
  const wasPopped = new Array<boolean>(nodeCount).fill(false);
  const deque: number[] = [source];
  distance[source] = 0;
  inQueue[source] = true;

  while (deque.length > 0) {
    const node = deque.shift()!;
    inQueue[node] = false;
    wasPopped[node] = true;

    for (const [neighbor, weight] of graph[node]) {
      const candidate = distance[node] + weight;
      if (candidate >= distance[neighbor]) {
        continue;
      }

      distance[neighbor] = candidate;
      if (inQueue[neighbor]) {
        continue;
      }

      if (wasPopped[neighbor]) {
        deque.unshift(neighbor);
      } else {
        deque.push(neighbor);
      }
      inQueue[neighbor] = true;
    }
  }

  return distance;
}
```

## Common Pitfalls

- Assuming the deque heuristic gives a strict worst-case improvement over Dijkstra.
- Re-enqueuing vertices blindly without tracking queue membership.
- Applying the method without understanding the graph's weight assumptions.
