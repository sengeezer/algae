## Why It Surfaces

This problem forces you to combine recursion with aggressive pruning. The key insight is that you only recurse on split positions when the two substrings have matching character inventories.

## Worked Example

`"great"` and `"rgeat"` are scramble-equivalent because `"great"` can be split as `"gr" + "eat"`, and `"gr"` can be swapped internally to form `"rg"` while `"eat"` stays the same.

```js
function scrambleStrings(left, right, memo = new Map()) {
  if (left.length !== right.length) {
    return false;
  }

  const key = `${left}|${right}`;
  if (memo.has(key)) {
    return memo.get(key);
  }

  if (left === right) {
    memo.set(key, true);
    return true;
  }

  const counts = new Map();
  for (let index = 0; index < left.length; index += 1) {
    counts.set(left[index], (counts.get(left[index]) ?? 0) + 1);
    counts.set(right[index], (counts.get(right[index]) ?? 0) - 1);
  }

  for (const value of counts.values()) {
    if (value !== 0) {
      memo.set(key, false);
      return false;
    }
  }

  for (let split = 1; split < left.length; split += 1) {
    const straight =
      scrambleStrings(left.slice(0, split), right.slice(0, split), memo) &&
      scrambleStrings(left.slice(split), right.slice(split), memo);

    const swapped =
      scrambleStrings(left.slice(0, split), right.slice(-split), memo) &&
      scrambleStrings(left.slice(split), right.slice(0, -split), memo);

    if (straight || swapped) {
      memo.set(key, true);
      return true;
    }
  }

  memo.set(key, false);
  return false;
}
```

export function scrambleStrings(
  left: string,
  right: string,
  memo = new Map<string, boolean>(),
): boolean {
  if (left.length !== right.length) {
    return false;
  }

  const key = `${left}|${right}`;
  if (memo.has(key)) {
    return memo.get(key)!;
  }

  if (left === right) {
    memo.set(key, true);
    return true;
  }

  const counts = new Map<string, number>();
  for (let index = 0; index < left.length; index += 1) {
    counts.set(left[index], (counts.get(left[index]) ?? 0) + 1);
    counts.set(right[index], (counts.get(right[index]) ?? 0) - 1);
  }

  for (const value of counts.values()) {
    if (value !== 0) {
      memo.set(key, false);
      return false;
    }
  }

  for (let split = 1; split < left.length; split += 1) {
    const straight =
      scrambleStrings(left.slice(0, split), right.slice(0, split), memo) &&
      scrambleStrings(left.slice(split), right.slice(split), memo);

    const swapped =
      scrambleStrings(left.slice(0, split), right.slice(-split), memo) &&
      scrambleStrings(left.slice(split), right.slice(0, -split), memo);

    if (straight || swapped) {
      memo.set(key, true);
      return true;
    }
  }

  memo.set(key, false);
  return false;
}

## Follow-up Questions

- Convert the recursive solution into interval-based dynamic programming.
- Return one valid split tree instead of only a boolean.
- Explain how the frequency-pruning step avoids unnecessary recursion.

## Common Pitfalls

- Recursing without first checking whether the character counts can match.
- Forgetting the swapped-subtree case.
- Omitting memoization and re-solving the same substring pairs repeatedly.
