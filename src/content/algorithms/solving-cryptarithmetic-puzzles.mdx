## Why It Surfaces

This is a classic constraint-search puzzle. The brute-force version tries digit assignments for every letter, but the useful interview move is to validate the addition one column at a time from right to left so bad carries fail early.

## Worked Example

For `SEND + MORE = MONEY`, the assignment `S=9, E=5, N=6, D=7, M=1, O=0, R=8, Y=2` works because `9567 + 1085 = 10652`. The key is that each letter keeps the same digit everywhere and neither `S` nor `M` is allowed to be `0`.

```js
function solvingCryptarithmeticPuzzles(addendA, addendB, total) {
  if (![addendA, addendB, total].every((value) => typeof value === 'string' && value.length > 0)) {
    throw new TypeError('All puzzle words must be non-empty strings');
  }

  const uniqueLetters = new Set([...addendA, ...addendB, ...total]);
  if (uniqueLetters.size > 10) {
    return null;
  }

  const leading = new Set([
    addendA[0],
    addendB[0],
    total[0],
  ]);

  const assignment = new Map();
  const usedDigits = new Set();

  function solveColumn(offset, row, carry) {
    if (offset === total.length) {
      return carry === 0;
    }

    const chars = [
      charFromRight(addendA, offset),
      charFromRight(addendB, offset),
      charFromRight(total, offset),
    ];

    if (row < 2) {
      const current = chars[row];
      if (current === null) {
        return solveColumn(offset, row + 1, carry);
      }

      if (assignment.has(current)) {
        return solveColumn(offset, row + 1, carry + assignment.get(current));
      }

      for (let digit = 0; digit <= 9; digit += 1) {
        if (usedDigits.has(digit)) {
          continue;
        }

        if (digit === 0 && leading.has(current)) {
          continue;
        }

        assignment.set(current, digit);
        usedDigits.add(digit);

        if (solveColumn(offset, row + 1, carry + digit)) {
          return true;
        }

        usedDigits.delete(digit);
        assignment.delete(current);
      }

      return false;
    }

    const resultChar = chars[2];
    const expectedDigit = carry % 10;
    const nextCarry = Math.floor(carry / 10);

    if (resultChar === null) {
      return expectedDigit === 0 && solveColumn(offset + 1, 0, nextCarry);
    }

    if (assignment.has(resultChar)) {
      return assignment.get(resultChar) === expectedDigit && solveColumn(offset + 1, 0, nextCarry);
    }

    if (usedDigits.has(expectedDigit)) {
      return false;
    }

    if (expectedDigit === 0 && leading.has(resultChar)) {
      return false;
    }

    assignment.set(resultChar, expectedDigit);
    usedDigits.add(expectedDigit);

    if (solveColumn(offset + 1, 0, nextCarry)) {
      return true;
    }

    usedDigits.delete(expectedDigit);
    assignment.delete(resultChar);
    return false;
  }

  if (!solveColumn(0, 0, 0)) {
    return null;
  }

  return {
    assignment: Object.fromEntries(assignment),
    values: {
      addendA: decodeWord(addendA, assignment),
      addendB: decodeWord(addendB, assignment),
      total: decodeWord(total, assignment),
    },
  };
}

function charFromRight(word, offset) {
  const index = word.length - 1 - offset;
  return index >= 0 ? word[index] : null;
}

function decodeWord(word, assignment) {
  return [...word].map((char) => assignment.get(char)).join('');
}
```

```ts
type CryptarithmSolution = {
  assignment: Record<string, number>;
  values: {
    addendA: string;
    addendB: string;
    total: string;
  };
};

export function solvingCryptarithmeticPuzzles(
  addendA: string,
  addendB: string,
  total: string,
): CryptarithmSolution | null {
  const uniqueLetters = new Set<string>([...addendA, ...addendB, ...total]);
  if (uniqueLetters.size > 10) {
    return null;
  }

  const leading = new Set<string>([addendA[0], addendB[0], total[0]]);
  const assignment = new Map<string, number>();
  const usedDigits = new Set<number>();

  const solveColumn = (offset: number, row: number, carry: number): boolean => {
    if (offset === total.length) {
      return carry === 0;
    }

    const chars = [
      charFromRight(addendA, offset),
      charFromRight(addendB, offset),
      charFromRight(total, offset),
    ];

    if (row < 2) {
      const current = chars[row];
      if (current === null) {
        return solveColumn(offset, row + 1, carry);
      }

      if (assignment.has(current)) {
        return solveColumn(offset, row + 1, carry + (assignment.get(current) as number));
      }

      for (let digit = 0; digit <= 9; digit += 1) {
        if (usedDigits.has(digit)) {
          continue;
        }

        if (digit === 0 && leading.has(current)) {
          continue;
        }

        assignment.set(current, digit);
        usedDigits.add(digit);

        if (solveColumn(offset, row + 1, carry + digit)) {
          return true;
        }

        usedDigits.delete(digit);
        assignment.delete(current);
      }

      return false;
    }

    const resultChar = chars[2];
    const expectedDigit = carry % 10;
    const nextCarry = Math.floor(carry / 10);

    if (resultChar === null) {
      return expectedDigit === 0 && solveColumn(offset + 1, 0, nextCarry);
    }

    if (assignment.has(resultChar)) {
      return (assignment.get(resultChar) as number) === expectedDigit
        && solveColumn(offset + 1, 0, nextCarry);
    }

    if (usedDigits.has(expectedDigit)) {
      return false;
    }

    if (expectedDigit === 0 && leading.has(resultChar)) {
      return false;
    }

    assignment.set(resultChar, expectedDigit);
    usedDigits.add(expectedDigit);

    if (solveColumn(offset + 1, 0, nextCarry)) {
      return true;
    }

    usedDigits.delete(expectedDigit);
    assignment.delete(resultChar);
    return false;
  };

  if (!solveColumn(0, 0, 0)) {
    return null;
  }

  return {
    assignment: Object.fromEntries(assignment),
    values: {
      addendA: decodeWord(addendA, assignment),
      addendB: decodeWord(addendB, assignment),
      total: decodeWord(total, assignment),
    },
  };
}

function charFromRight(word: string, offset: number): string | null {
  const index = word.length - 1 - offset;
  return index >= 0 ? word[index] : null;
}

function decodeWord(word: string, assignment: Map<string, number>): string {
  return [...word].map((char) => assignment.get(char)).join('');
}
```

## Follow-up Questions

- How would you generalize this to more than two addends?
- Why is column-wise validation much stronger than checking only after a full assignment?
- What extra pruning would you add if the puzzle had many repeated letters in one column?

## Common Pitfalls

- Allowing a leading character to map to `0`.
- Checking the equation only after assigning every letter.
- Forgetting that the same letter can appear in multiple columns and must keep one digit globally.
