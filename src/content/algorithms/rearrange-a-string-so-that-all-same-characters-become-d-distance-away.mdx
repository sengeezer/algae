## Why It Surfaces

This is the adjacency problem with a larger cooldown window. The greedy move is still to use the most frequent available character first, but now a used character must be held out of the heap until it is legal to place again.

## Worked Example

For `s = "aacbbc"` and `d = 3`, placing the most frequent available characters with a three-step cooldown can produce `abcabc`. Each repeated character is exactly three positions away from its next occurrence, so the arrangement is valid.

function rearrangeAStringSoThatAllSameCharactersBecomeDDistanceAway(
  text,
  distance,
) {
  if (distance <= 1 || text.length <= 1) {
    return text;
  }

  const frequencies = new Map();

  for (const char of text) {
    frequencies.set(char, (frequencies.get(char) ?? 0) + 1);
  }

  const heap = [];

  function comesBefore(left, right) {
    return left.count > right.count || (left.count === right.count && left.char < right.char);
  }

  function swap(left, right) {
    [heap[left], heap[right]] = [heap[right], heap[left]];
  }

  function siftUp(index) {
    let child = index;

    while (child > 0) {
      const parent = Math.floor((child - 1) / 2);

      if (comesBefore(heap[parent], heap[child])) {
        break;
      }

      swap(parent, child);
      child = parent;
    }
  }

  function siftDown(index) {
    let parent = index;

    while (true) {
      let best = parent;
      const left = parent * 2 + 1;
      const right = parent * 2 + 2;

      if (left < heap.length && comesBefore(heap[left], heap[best])) {
        best = left;
      }

      if (right < heap.length && comesBefore(heap[right], heap[best])) {
        best = right;
      }

      if (best === parent) {
        break;
      }

      swap(parent, best);
      parent = best;
    }
  }

  function push(value) {
    heap.push(value);
    siftUp(heap.length - 1);
  }

  function pop() {
    const top = heap[0];
    const last = heap.pop();

    if (heap.length > 0 && last !== undefined) {
      heap[0] = last;
      siftDown(0);
    }

    return top;
  }

  for (const [char, count] of frequencies.entries()) {
    push({ char, count });
  }

  const cooldown = [];
  let cooldownHead = 0;
  let result = '';

  for (let step = 0; step < text.length; step += 1) {
    while (
      cooldownHead < cooldown.length &&
      cooldown[cooldownHead].releaseStep <= step
    ) {
      push(cooldown[cooldownHead]);
      cooldownHead += 1;
    }

    if (heap.length === 0) {
      return '';
    }

    const current = pop();
    result += current.char;
    current.count -= 1;

    if (current.count > 0) {
      cooldown.push({
        char: current.char,
        count: current.count,
        releaseStep: step + distance,
      });
    }
  }

  return result;
}

    return text;
  }

  }

    let child = index;

      const parent = Math.floor((child - 1) / 2);

        break;

      child = parent;

## Follow-up Questions

- Return one of the placement schedules that proves the arrangement impossible.
- Compare the `d = 2` case with the simpler no-adjacent-equals variant.
- Adapt the routine if different characters have different cooldown distances.

## Common Pitfalls

- Reusing a character before its cooldown expires.
- Returning any prefix built so far when the heap runs dry before the output is complete.
- Claiming O(1) queue operations while using `shift()` on a JavaScript array.
