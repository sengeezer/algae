## Why It Surfaces

Intro sort is what you discuss when the interviewer stops caring about one elegant idea and starts caring about robust engineering. It combines three familiar algorithms because each one is strong in a different regime.

## Worked Example

On a large random array, intro sort behaves mostly like quicksort. If unlucky pivots make the recursion depth grow too far, it switches to heapsort for that segment, and tiny partitions are handled with insertion sort because the constant factors are better there.

function introSort(values) {
  if (!Array.isArray(values)) {
    throw new TypeError('Values must be an array of numbers');
  }

  const items = [...values];
  const threshold = 16;
  const maxDepth = 2 * Math.floor(Math.log2(items.length || 1));

  function swap(left, right) {
    [items[left], items[right]] = [items[right], items[left]];
  }

  function insertionSort(left, right) {
    for (let index = left + 1; index <= right; index += 1) {
      const current = items[index];
      let position = index - 1;

      while (position >= left && items[position] > current) {
        items[position + 1] = items[position];
        position -= 1;
      }

      items[position + 1] = current;
    }
  }

  function partition(left, right) {
    const pivot = items[Math.floor((left + right) / 2)];
    let low = left;
    let high = right;

    while (low <= high) {
      while (items[low] < pivot) {
        low += 1;
      }

      while (items[high] > pivot) {
        high -= 1;
      }

      if (low <= high) {
        swap(low, high);
        low += 1;
        high -= 1;
      }
    }

    return low;
  }

  function heapSort(left, right) {
    const size = right - left + 1;

    function valueAt(index) {
      return items[left + index];
    }

    function swapHeap(first, second) {
      swap(left + first, left + second);
    }

    function siftDown(root, end) {
      while (root * 2 + 1 <= end) {
        const child = root * 2 + 1;
        let candidate = root;

        if (valueAt(candidate) < valueAt(child)) {
          candidate = child;
        }

        if (child + 1 <= end && valueAt(candidate) < valueAt(child + 1)) {
          candidate = child + 1;
        }

        if (candidate === root) {
          return;
        }

        swapHeap(root, candidate);
        root = candidate;
      }
    }

    for (let root = Math.floor((size - 2) / 2); root >= 0; root -= 1) {
      siftDown(root, size - 1);
    }

    for (let end = size - 1; end > 0; end -= 1) {
      swapHeap(0, end);
      siftDown(0, end - 1);
    }
  }

  function sort(left, right, depthLimit) {
    if (right - left + 1 <= threshold) {
      insertionSort(left, right);
      return;
    }

    if (depthLimit === 0) {
      heapSort(left, right);
      return;
    }

    const split = partition(left, right);

    if (left < split - 1) {
      sort(left, split - 1, depthLimit - 1);
    }

    if (split < right) {
      sort(split, right, depthLimit - 1);
    }
  }

  if (items.length > 1) {
    sort(0, items.length - 1, maxDepth);
  }

  return items;
}

## Follow-up Questions

- Explain why the depth limit keeps quicksort from degrading catastrophically.
- Compare intro sort with tim sort as two different kinds of hybrid production sort.
- Replace the pivot rule with median-of-three and discuss the effect.

## Common Pitfalls

- Claiming intro sort is just quicksort with a better pivot rule.
- Forgetting that small partitions are often faster with insertion sort.
- Omitting the heapsort fallback and losing the worst-case guarantee.
