## Why It Surfaces

The prompt looks like a greedy array-editing problem, but the real question is whether a candidate window can be flattened to one value within budget. Because decrements are the only allowed operation, the target must be the interval minimum, so every check reduces to `sum(window) - min(window) * length(window)`.

## Worked Example

For `[1, 7, 3, 4, 5, 6]` with `K = 6`, the best window is `[3, 4, 5, 6]`. Its minimum is `3`, and the total decrements needed are `(3 + 4 + 5 + 6) - 4 * 3 = 6`, so the answer is `4`.

class RangeMinSegmentTree {
  constructor(values) {
    if (!Array.isArray(values) || values.length === 0) {
      throw new RangeError('Values must be a non-empty array');
    }

    this.values = values.slice();
    this.size = values.length;
    this.tree = new Array(this.size * 4).fill(Number.POSITIVE_INFINITY);
    this.build(1, 0, this.size - 1);
  }

  build(node, left, right) {
    if (left === right) {
      this.tree[node] = this.values[left];
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = Math.min(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  query(queryLeft, queryRight) {
    return this.queryNode(1, 0, this.size - 1, queryLeft, queryRight);
  }

  queryNode(node, left, right, queryLeft, queryRight) {
    if (queryRight < left || right < queryLeft) {
      return Number.POSITIVE_INFINITY;
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return Math.min(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }
}

function maximizeLengthOfLongestSubarrayConsistingOfSameElementsByAtMostKDecrements(
  values,
  maxDecrements,
) {
  if (!Array.isArray(values) || values.length === 0) {
    return 0;
  }

  if (!Number.isInteger(maxDecrements) || maxDecrements < 0) {
    throw new RangeError('K must be a non-negative integer');
  }

  const prefixSums = new Array(values.length + 1).fill(0);
  for (let index = 0; index < values.length; index += 1) {
    prefixSums[index + 1] = prefixSums[index] + values[index];
  }

  const minTree = new RangeMinSegmentTree(values);

  const canFlatten = (left, right) => {
    const minimum = minTree.query(left, right);
    const sum = prefixSums[right + 1] - prefixSums[left];
    const length = right - left + 1;
    return sum - minimum * length <= maxDecrements;
  };

  let best = 1;

  for (let left = 0; left < values.length; left += 1) {
    let low = left;
    let high = values.length - 1;
    let answer = left;

    while (low <= high) {
      const mid = Math.floor((low + high) / 2);
      if (canFlatten(left, mid)) {
        answer = mid;
        low = mid + 1;
      } else {
        high = mid - 1;
      }
    }

    best = Math.max(best, answer - left + 1);
  }

  return best;
}

## Common Pitfalls

- Forgetting that only decrements are allowed.
- Recomputing the minimum of every candidate range from scratch.
- Binary-searching window lengths without a fast feasibility check.
