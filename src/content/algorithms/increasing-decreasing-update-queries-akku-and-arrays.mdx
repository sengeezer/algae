## Why It Surfaces

Monotonicity over a concatenated range depends only on three things: whether the left half already has the property, whether the right half has it, and whether the boundary pair respects the required order. That makes the merge rule small and exact.

## Worked Example

For `[1, 5, 7, 4, 3, 5, 9]`, the query `[1, 3]` is non-decreasing, so the answer is `0`. After updating index `7` to `4`, the query `[6, 7]` becomes `[5, 4]`, which is non-increasing, so the answer is `1`.

function increasingDecreasingUpdateQueriesAkkuAndArrays(values, queries) {
  const tree = new TrendSegmentTree(values);
  const answers = [];

  for (const query of queries) {
    if (!Array.isArray(query) || query.length !== 3) {
      throw new RangeError('Each query must have exactly three integers');
    }

    const [type, first, second] = query;

    if (type === 1) {
      tree.update(first - 1, second);
    } else if (type === 2) {
      const result = tree.query(first - 1, second - 1);
      answers.push(classifyTrend(result));
    } else {
      throw new RangeError('Query type must be 1 or 2');
    }
  }

  return answers;
}

class TrendSegmentTree {
  constructor(values) {
    if (!Array.isArray(values) || values.length === 0) {
      throw new RangeError('Values must be a non-empty array');
    }

    this.values = values.map((value) => {
      if (!Number.isFinite(value)) {
        throw new RangeError('Array values must be finite numbers');
      }

      return value;
    });

    this.tree = new Array(this.values.length * 4).fill(null);
    this.build(1, 0, this.values.length - 1);
  }

  update(index, value) {
    if (!Number.isInteger(index) || index < 0 || index >= this.values.length || !Number.isFinite(value)) {
      throw new RangeError('Update must target a valid zero-based index with a finite value');
    }

    this.values[index] = value;
    this.updateNode(1, 0, this.values.length - 1, index, value);
  }

  query(left, right) {
    if (!Number.isInteger(left) || !Number.isInteger(right) || left < 0 || right >= this.values.length || left > right) {
      throw new RangeError('Query must be a valid zero-based inclusive range');
    }

    return this.queryNode(1, 0, this.values.length - 1, left, right);
  }

  build(node, left, right) {
    if (left === right) {
      const value = this.values[left];
      this.tree[node] = leafTrendNode(value);
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = mergeTrendNodes(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  updateNode(node, left, right, index, value) {
    if (left === right) {
      this.tree[node] = leafTrendNode(value);
      return;
    }

    const mid = Math.floor((left + right) / 2);
    if (index <= mid) {
      this.updateNode(node * 2, left, mid, index, value);
    } else {
      this.updateNode(node * 2 + 1, mid + 1, right, index, value);
    }

    this.tree[node] = mergeTrendNodes(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  queryNode(node, left, right, queryLeft, queryRight) {
    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    if (queryRight <= mid) {
      return this.queryNode(node * 2, left, mid, queryLeft, queryRight);
    }

    if (queryLeft > mid) {
      return this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight);
    }

    return mergeTrendNodes(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }
}

function leafTrendNode(value) {
  return {
    leftValue: value,
    rightValue: value,
    isNonDecreasing: true,
    isNonIncreasing: true,
  };
}

function mergeTrendNodes(left, right) {
  return {
    leftValue: left.leftValue,
    rightValue: right.rightValue,
    isNonDecreasing:
      left.isNonDecreasing && right.isNonDecreasing && left.rightValue <= right.leftValue,
    isNonIncreasing:
      left.isNonIncreasing && right.isNonIncreasing && left.rightValue >= right.leftValue,
  };
}

function classifyTrend(node) {
  if (node.isNonDecreasing && !node.isNonIncreasing) {
    return 0;
  }

  if (node.isNonIncreasing && !node.isNonDecreasing) {
    return 1;
  }

  return -1;
}

type TrendNode = {
  leftValue: number;
  rightValue: number;
  isNonDecreasing: boolean;
  isNonIncreasing: boolean;
};

class TrendSegmentTree {
  private readonly values: number[];
  private readonly tree: TrendNode[];

  constructor(values: number[]) {
    if (values.length === 0) {
      throw new RangeError('Values must be a non-empty array');
    }

    this.values = values.map((value) => {
      if (!Number.isFinite(value)) {
        throw new RangeError('Array values must be finite numbers');
      }

      return value;
    });

    this.tree = new Array<TrendNode>(this.values.length * 4);
    this.build(1, 0, this.values.length - 1);
  }

  update(index: number, value: number): void {
    if (!Number.isInteger(index) || index < 0 || index >= this.values.length || !Number.isFinite(value)) {
      throw new RangeError('Update must target a valid zero-based index with a finite value');
    }

    this.values[index] = value;
    this.updateNode(1, 0, this.values.length - 1, index, value);
  }

  query(left: number, right: number): TrendNode {
    if (!Number.isInteger(left) || !Number.isInteger(right) || left < 0 || right >= this.values.length || left > right) {
      throw new RangeError('Query must be a valid zero-based inclusive range');
    }

    return this.queryNode(1, 0, this.values.length - 1, left, right);
  }

  private build(node: number, left: number, right: number): void {
    if (left === right) {
      this.tree[node] = leafTrendNode(this.values[left]);
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = mergeTrendNodes(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  private updateNode(node: number, left: number, right: number, index: number, value: number): void {
    if (left === right) {
      this.tree[node] = leafTrendNode(value);
      return;
    }

    const mid = Math.floor((left + right) / 2);
    if (index <= mid) {
      this.updateNode(node * 2, left, mid, index, value);
    } else {
      this.updateNode(node * 2 + 1, mid + 1, right, index, value);
    }

    this.tree[node] = mergeTrendNodes(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  private queryNode(
    node: number,
    left: number,
    right: number,
    queryLeft: number,
    queryRight: number,
  ): TrendNode {
    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    if (queryRight <= mid) {
      return this.queryNode(node * 2, left, mid, queryLeft, queryRight);
    }

    if (queryLeft > mid) {
      return this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight);
    }

    return mergeTrendNodes(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }
}

  return {
    leftValue: value,
    rightValue: value,
    isNonDecreasing: true,
    isNonIncreasing: true,
  };
}

function mergeTrendNodes(left: TrendNode, right: TrendNode): TrendNode {

    leftValue: left.leftValue,
    rightValue: right.rightValue,

## Common Pitfalls

- Forgetting that equal adjacent values make a range both non-decreasing and non-increasing.
- Trying to classify a merged range without checking the boundary pair between children.
- Using zero-based indices directly even though the source problem statements use one-based queries.
