## Why It Surfaces

This is the classic example of carrying extra state through merge sort. Once both halves are individually sorted, every time the merge step prefers `right[j]` over `left[i]`, that one comparison certifies inversions with every remaining value from `left[i]` onward.

## Worked Example

For `[4, 3, 2, 1]`, the left half contributes `1` inversion, the right half contributes `1`, and the final merge contributes `4` more because both `1` and `2` jump ahead of two remaining left-half values. The total is `6`.

function countInversionsOfAnArray(values) {
  if (!Array.isArray(values)) {
    throw new TypeError('Values must be an array');
  }

  const items = values.map((value) => {
    if (!Number.isFinite(value)) {
      throw new RangeError('Array values must be finite numbers');
    }

    return value;
  });

  if (items.length <= 1) {
    return 0;
  }

  const buffer = new Array(items.length);
  return sortAndCount(items, buffer, 0, items.length - 1);
}

function sortAndCount(values, buffer, left, right) {
  if (left >= right) {
    return 0;
  }

  const mid = Math.floor((left + right) / 2);
  let inversions = 0;

  inversions += sortAndCount(values, buffer, left, mid);
  inversions += sortAndCount(values, buffer, mid + 1, right);
  inversions += mergeAndCount(values, buffer, left, mid, right);

  return inversions;
}

function mergeAndCount(values, buffer, left, mid, right) {
  let leftIndex = left;
  let rightIndex = mid + 1;
  let writeIndex = left;
  let inversions = 0;

  while (leftIndex <= mid && rightIndex <= right) {
    if (values[leftIndex] <= values[rightIndex]) {
      buffer[writeIndex++] = values[leftIndex++];
    } else {
      buffer[writeIndex++] = values[rightIndex++];
      inversions += mid - leftIndex + 1;
    }
  }

  while (leftIndex <= mid) {
    buffer[writeIndex++] = values[leftIndex++];
  }

  while (rightIndex <= right) {
    buffer[writeIndex++] = values[rightIndex++];
  }

  for (let index = left; index <= right; index += 1) {
    values[index] = buffer[index];
  }

  return inversions;
}

## Common Pitfalls

- Forgetting that only `arr[i] > arr[j]` counts, so equal values are not inversions.
- Adding only `1` inversion on a right-half win instead of all remaining left-half items.
- Mutating the caller's array accidentally when the function is expected to be pure.
