## Why It Surfaces

The basic segment tree handles point updates well, but range updates can degenerate into too many recursive calls. Lazy propagation fixes that by marking an interval as having pending work and pushing that work to children only when a later operation needs to inspect them.

## Worked Example

For `[1, 3, 5, 7, 9, 11]`, adding `10` over `[1, 3]` changes the logical array to `[1, 13, 15, 17, 9, 11]`. A query over `[1, 4]` then returns `54` even though not every descendant was updated eagerly at the moment the range-add operation arrived.

class LazyPropagationSegmentTree {
  constructor(values) {
    if (!Array.isArray(values) || values.length === 0) {
      throw new RangeError('Values must be a non-empty array');
    }

    this.size = values.length;
    this.tree = new Array(this.size * 4).fill(0);
    this.lazy = new Array(this.size * 4).fill(0);
    this.build(values, 1, 0, this.size - 1);
  }

  build(values, node, left, right) {
    if (left === right) {
      this.tree[node] = values[left];
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(values, node * 2, left, mid);
    this.build(values, node * 2 + 1, mid + 1, right);
    this.tree[node] = this.tree[node * 2] + this.tree[node * 2 + 1];
  }

  rangeAdd(updateLeft, updateRight, delta) {
    this.validateRange(updateLeft, updateRight);
    if (!Number.isFinite(delta)) {
      throw new TypeError('Delta must be a finite number');
    }

    this.rangeAddNode(1, 0, this.size - 1, updateLeft, updateRight, delta);
  }

  query(queryLeft, queryRight) {
    this.validateRange(queryLeft, queryRight);
    return this.queryNode(1, 0, this.size - 1, queryLeft, queryRight);
  }

  apply(node, left, right, delta) {
    this.tree[node] += (right - left + 1) * delta;
    this.lazy[node] += delta;
  }

  push(node, left, right) {
    const pending = this.lazy[node];
    if (pending === 0) {
      return;
    }

    if (left !== right) {
      const mid = Math.floor((left + right) / 2);
      this.apply(node * 2, left, mid, pending);
      this.apply(node * 2 + 1, mid + 1, right, pending);
    }

    this.lazy[node] = 0;
  }

  rangeAddNode(node, left, right, updateLeft, updateRight, delta) {
    if (updateRight < left || right < updateLeft) {
      return;
    }

    if (updateLeft <= left && right <= updateRight) {
      this.apply(node, left, right, delta);
      return;
    }

    this.push(node, left, right);
    const mid = Math.floor((left + right) / 2);
    this.rangeAddNode(node * 2, left, mid, updateLeft, updateRight, delta);
    this.rangeAddNode(node * 2 + 1, mid + 1, right, updateLeft, updateRight, delta);
    this.tree[node] = this.tree[node * 2] + this.tree[node * 2 + 1];
  }

  queryNode(node, left, right, queryLeft, queryRight) {
    if (queryRight < left || right < queryLeft) {
      return 0;
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    this.push(node, left, right);
    const mid = Math.floor((left + right) / 2);
    return (
      this.queryNode(node * 2, left, mid, queryLeft, queryRight)
      + this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight)
    );
  }

  validateRange(queryLeft, queryRight) {
    if (
      !Number.isInteger(queryLeft)
      || !Number.isInteger(queryRight)
      || queryLeft < 0
      || queryRight < queryLeft
      || queryRight >= this.size
    ) {
      throw new RangeError('Query range is invalid');
    }
  }
}

export class LazyPropagationSegmentTree {
  private readonly size: number;
  private readonly tree: number[];
  private readonly lazy: number[];

  constructor(values: number[]) {
    if (values.length === 0) {
      throw new RangeError('Values must be a non-empty array');
    }

    this.size = values.length;
    this.tree = new Array<number>(this.size * 4).fill(0);
    this.lazy = new Array<number>(this.size * 4).fill(0);
    this.build(values, 1, 0, this.size - 1);
  }

  rangeAdd(updateLeft: number, updateRight: number, delta: number): void {
    this.validateRange(updateLeft, updateRight);
    this.rangeAddNode(1, 0, this.size - 1, updateLeft, updateRight, delta);
  }

  query(queryLeft: number, queryRight: number): number {
    this.validateRange(queryLeft, queryRight);
    return this.queryNode(1, 0, this.size - 1, queryLeft, queryRight);
  }

  private build(values: number[], node: number, left: number, right: number): void {
    if (left === right) {
      this.tree[node] = values[left];
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(values, node * 2, left, mid);
    this.build(values, node * 2 + 1, mid + 1, right);
    this.tree[node] = this.tree[node * 2] + this.tree[node * 2 + 1];
  }

  private apply(node: number, left: number, right: number, delta: number): void {
    this.tree[node] += (right - left + 1) * delta;
    this.lazy[node] += delta;
  }

  private push(node: number, left: number, right: number): void {
    const pending = this.lazy[node];
    if (pending === 0) {
      return;
    }

    if (left !== right) {
      const mid = Math.floor((left + right) / 2);
      this.apply(node * 2, left, mid, pending);
      this.apply(node * 2 + 1, mid + 1, right, pending);
    }

    this.lazy[node] = 0;
  }

  private rangeAddNode(
    node: number,
    left: number,
    right: number,
    updateLeft: number,
    updateRight: number,
    delta: number,
  ): void {
    if (updateRight < left || right < updateLeft) {
      return;
    }

    if (updateLeft <= left && right <= updateRight) {
      this.apply(node, left, right, delta);
      return;
    }

    this.push(node, left, right);
    const mid = Math.floor((left + right) / 2);
    this.rangeAddNode(node * 2, left, mid, updateLeft, updateRight, delta);
    this.rangeAddNode(node * 2 + 1, mid + 1, right, updateLeft, updateRight, delta);
    this.tree[node] = this.tree[node * 2] + this.tree[node * 2 + 1];
  }

  private queryNode(
    node: number,
    left: number,
    right: number,
    queryLeft: number,
    queryRight: number,
  ): number {
    if (queryRight < left || right < queryLeft) {
      return 0;
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    this.push(node, left, right);
    const mid = Math.floor((left + right) / 2);
    return (
      this.queryNode(node * 2, left, mid, queryLeft, queryRight)
      + this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight)
    );
  }

  private validateRange(queryLeft: number, queryRight: number): void {
    if (
      !Number.isInteger(queryLeft)
      || !Number.isInteger(queryRight)
      || queryLeft < 0
      || queryRight < queryLeft
      || queryRight >= this.size
    ) {
      throw new RangeError('Query range is invalid');
    }
  }
}

## Follow-up Questions

- Adapt the tree for range assignment instead of range addition.
- Explain why the pending value must be applied before descending into a partially overlapping node.
- Compare this with a difference array when there are no online queries.

## Common Pitfalls

- Forgetting that a segment sum changes by `segmentLength * delta`, not just `delta`.
- Returning stale values because pending updates were never pushed on partial overlap.
- Reusing point-update logic and accidentally turning a logarithmic range update back into linear work.
