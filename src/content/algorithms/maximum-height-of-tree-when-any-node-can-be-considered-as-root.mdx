## Why It Surfaces

This is the classic rerooting move. A single subtree height is easy, but once every node is allowed to be the root, each node needs to know both the best path going down into its own children and the best alternative path coming from its parent side.

## Worked Example

For `nodeCount = 5` and `edges = [[0, 1], [1, 2], [1, 3], [3, 4]]`, the answer is `3`. Rooting the tree at either endpoint of the longest path gives height `3`, while the more central nodes produce smaller heights.

function maximumHeightOfTreeWhenAnyNodeCanBeConsideredAsRoot(nodeCount, edges) {
  if (!Number.isInteger(nodeCount) || nodeCount < 0) {
    throw new RangeError('nodeCount must be a non-negative integer');
  }

  if (!Array.isArray(edges)) {
    throw new TypeError('edges must be an array of [from, to] pairs');
  }

  if (nodeCount <= 1) {
    return 0;
  }

  const graph = Array.from({ length: nodeCount }, () => []);

  for (const edge of edges) {
    if (!Array.isArray(edge) || edge.length !== 2) {
      throw new TypeError('edges must be an array of [from, to] pairs');
    }

    const [from, to] = edge;
    if (
      !Number.isInteger(from)
      || !Number.isInteger(to)
      || from < 0
      || to < 0
      || from >= nodeCount
      || to >= nodeCount
    ) {
      throw new RangeError('edge endpoints must be valid node indices');
    }

    graph[from].push(to);
    graph[to].push(from);
  }

  const down = new Array(nodeCount).fill(0);
  const up = new Array(nodeCount).fill(0);

  function dfsDown(node, parent) {
    for (const neighbor of graph[node]) {
      if (neighbor === parent) {
        continue;
      }

      down[node] = Math.max(down[node], 1 + dfsDown(neighbor, node));
    }

    return down[node];
  }

  function dfsUp(node, parent) {
    let bestChild = -1;
    let secondBestChild = -1;

    for (const neighbor of graph[node]) {
      if (neighbor === parent) {
        continue;
      }

      const candidate = 1 + down[neighbor];
      if (candidate > bestChild) {
        secondBestChild = bestChild;
        bestChild = candidate;
      } else if (candidate > secondBestChild) {
        secondBestChild = candidate;
      }
    }

    for (const neighbor of graph[node]) {
      if (neighbor === parent) {
        continue;
      }

      const childPath = 1 + down[neighbor];
      const useSibling = childPath === bestChild ? secondBestChild : bestChild;
      up[neighbor] = 1 + Math.max(up[node], useSibling);
      dfsUp(neighbor, node);
    }
  }

  dfsDown(0, -1);
  dfsUp(0, -1);

  let answer = 0;
  for (let node = 0; node < nodeCount; node += 1) {
    answer = Math.max(answer, Math.max(down[node], up[node]));
  }

  return answer;
}

## Follow-up Questions

- Return the height for every possible root instead of only the maximum over them.
- Explain why the tree diameter gives the same final maximum height.
- Adapt the rerooting state when nodes have weights instead of unit-height edges.

## Common Pitfalls

- Recomputing a full DFS from every node and missing the rerooting shortcut.
- Forgetting to keep both the best and second-best child contributions when moving the root.
- Mixing node-count and edge-count definitions of height.
