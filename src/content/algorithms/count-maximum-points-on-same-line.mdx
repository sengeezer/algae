## Why It Surfaces

The real trick here is not computing slope once, but comparing many slopes exactly. If you store raw floating-point ratios, tiny precision differences break collinearity grouping, so the better move is to normalize the direction `(dy, dx)` with a greatest common divisor and a fixed sign convention.

## Worked Example

For points `[[-1, 1], [0, 0], [1, 1], [2, 2], [3, 3], [3, 4]]`, the largest collinear set is `[0, 0]`, `[1, 1]`, `[2, 2]`, `[3, 3]`, so the answer is `4`.

function countMaximumPointsOnSameLine(points) {
  function gcd(left, right) {
    let a = Math.abs(left);
    let b = Math.abs(right);

    while (b !== 0) {
      [a, b] = [b, a % b];
    }

    return a;
  }

  function slopeKey(anchor, candidate) {
    let dy = candidate[1] - anchor[1];
    let dx = candidate[0] - anchor[0];

    if (dx === 0) {
      return "1/0";
    }

    if (dy === 0) {
      return "0/1";
    }

    const divisor = gcd(dy, dx);
    dy /= divisor;
    dx /= divisor;

    if (dx < 0) {
      dy = -dy;
      dx = -dx;
    }

    return `${dy}/${dx}`;
  }

  if (points.length < 2) {
    return points.length;
  }

  let best = 1;

  for (let index = 0; index < points.length; index += 1) {
    const slopeCounts = new Map();
    let overlaps = 0;
    let localBest = 0;

    for (let next = index + 1; next < points.length; next += 1) {
      if (
        points[index][0] === points[next][0]
        && points[index][1] === points[next][1]
      ) {
        overlaps += 1;
        continue;
      }

      const key = slopeKey(points[index], points[next]);
      const count = (slopeCounts.get(key) ?? 0) + 1;
      slopeCounts.set(key, count);
      localBest = Math.max(localBest, count);
    }

    best = Math.max(best, localBest + overlaps + 1);
  }

  return best;
}

## Common Pitfalls

- Letting floating-point rounding split one true line across multiple keys.
- Forgetting to treat duplicate points separately from shared slopes.
- Mishandling vertical lines when `dx = 0`.
