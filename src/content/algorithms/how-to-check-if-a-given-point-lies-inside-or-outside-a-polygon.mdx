## Why It Surfaces

This is the standard point-in-polygon interview question. The stable mental model is to cast a ray to the right, count edge crossings, and short-circuit to `true` when the point lies exactly on an edge.

## Worked Example

For the square with vertices `(0, 0)`, `(10, 0)`, `(10, 10)`, and `(0, 10)`, the point `(5, 5)` is inside because a ray to the right crosses the boundary once, while `(15, 5)` is outside because it crosses zero times.

```js
function howToCheckIfAGivenPointLiesInsideOrOutsideAPolygon(point, polygon) {
  if (!Array.isArray(polygon) || polygon.length < 3) {
    throw new RangeError("Polygon must contain at least three vertices");
  }

  const epsilon = 1e-9;

  function cross(origin, first, second) {
    return (first.x - origin.x) * (second.y - origin.y) - (first.y - origin.y) * (second.x - origin.x);
  }

  function isOnSegment(candidate, start, end) {
    if (Math.abs(cross(start, end, candidate)) > epsilon) {
      return false;
    }

    return (
      candidate.x >= Math.min(start.x, end.x) - epsilon &&
      candidate.x <= Math.max(start.x, end.x) + epsilon &&
      candidate.y >= Math.min(start.y, end.y) - epsilon &&
      candidate.y <= Math.max(start.y, end.y) + epsilon
    );
  }

  let inside = false;

  for (let index = 0; index < polygon.length; index += 1) {
    const current = polygon[index];
    const next = polygon[(index + 1) % polygon.length];

    if (isOnSegment(point, current, next)) {
      return true;
    }

    const crossesHorizontalBand = (current.y > point.y) !== (next.y > point.y);
    if (!crossesHorizontalBand) {
      continue;
    }

    const intersectionX = current.x + ((next.x - current.x) * (point.y - current.y)) / (next.y - current.y);
    if (intersectionX >= point.x - epsilon) {
      inside = !inside;
    }
  }

  return inside;
}
```

## Common Pitfalls

- Double-counting a ray that passes exactly through a polygon vertex.
- Forgetting to treat boundary points as inside.
- Using a horizontal-band test that includes both endpoints and flips parity twice on shared vertices.
