## Why It Surfaces

This is the graph version of Number of Islands. Instead of starting from every node blindly, you only launch a traversal when you find a node that no earlier traversal has already absorbed.

## JavaScript

```js
function connectedComponents(graph) {
  const visited = new Set();
  const components = [];

  function dfs(node, component) {
    visited.add(node);
    component.push(node);

    for (const neighbor of graph[node]) {
      if (!visited.has(neighbor)) {
        dfs(neighbor, component);
      }
    }
  }

  for (const node of Object.keys(graph)) {
    if (visited.has(node)) {
      continue;
    }

    const component = [];
    dfs(node, component);
    components.push(component);
  }

  return components;
}
```

## TypeScript

```ts
export function countConnectedComponents(graph: number[][]): number {
  const visited = new Array<boolean>(graph.length).fill(false);
  let components = 0;

  function dfs(node: number): void {
    visited[node] = true;

    for (const neighbor of graph[node]) {
      if (!visited[neighbor]) {
        dfs(neighbor);
      }
    }
  }

  for (let node = 0; node < graph.length; node += 1) {
    if (visited[node]) {
      continue;
    }

    components += 1;
    dfs(node);
  }

  return components;
}
```

## Interview Notes

- Say explicitly that each DFS or BFS from an unvisited node discovers exactly one component.
- Mention union-find as an alternative if edges are streamed incrementally.
- Clarify whether the graph is undirected, because strong connectivity is a different problem on directed graphs.
