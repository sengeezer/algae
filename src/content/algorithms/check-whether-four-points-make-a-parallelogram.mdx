## Why It Surfaces

The coordinates are unordered, so checking side vectors directly is awkward. The cleaner invariant is that the diagonals of a parallelogram bisect each other, which means exactly one midpoint appears twice among the six pairwise midpoints, provided the shape is not degenerate.

## Worked Example

The points `(0, 0)`, `(4, 0)`, `(1, 3)`, and `(5, 3)` form a parallelogram because the two diagonals share the same midpoint and the points are not collinear.

function checkWhetherFourPointsMakeAParallelogram(points) {
  if (
    !Array.isArray(points)
    || points.length !== 4
    || !points.every((point) => Array.isArray(point) && point.length === 2 && point.every(Number.isFinite))
  ) {
    throw new RangeError("Points must be an array of four coordinate pairs");
  }

  const uniquePointCount = new Set(points.map(([x, y]) => `${x},${y}`)).size;
  if (uniquePointCount !== 4) {
    return false;
  }

  const [firstPoint] = points;
  const secondPoint = points.find(([x, y]) => x !== firstPoint[0] || y !== firstPoint[1]);
  if (!secondPoint) {
    return false;
  }

  const directionX = secondPoint[0] - firstPoint[0];
  const directionY = secondPoint[1] - firstPoint[1];
  const allCollinear = points.every(([x, y]) => directionX * (y - firstPoint[1]) === directionY * (x - firstPoint[0]));
  if (allCollinear) {
    return false;
  }

  const midpointCounts = new Map();
  for (let left = 0; left < 4; left += 1) {
    for (let right = left + 1; right < 4; right += 1) {
      const key = `${points[left][0] + points[right][0]},${points[left][1] + points[right][1]}`;
      midpointCounts.set(key, (midpointCounts.get(key) ?? 0) + 1);
    }
  }

  return [...midpointCounts.values()].filter((count) => count === 2).length === 1;
}

## Common Pitfalls

- Assuming the input points are already listed in polygon order.
- Forgetting to reject duplicate points or fully collinear inputs.
- Comparing floating midpoints directly instead of comparing doubled midpoint coordinates.
