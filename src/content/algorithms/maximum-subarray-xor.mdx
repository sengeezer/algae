## Why It Surfaces

This problem is a classic example of using a trie for something other than strings. Once you rewrite every subarray XOR as `prefix[j] ^ prefix[i - 1]`, the whole task becomes: for each prefix, find a previous prefix that maximizes XOR with it.

## Worked Example

For `[1, 2, 3, 4]`, the subarray `[3, 4]` has XOR `7`, which is the maximum among all contiguous subarrays, so the function returns `7`.

```js
function createTrieNode() {
  return { children: [null, null] };
}

function insertTrieValue(root, value) {
  let node = root;

  for (let bit = 31; bit >= 0; bit -= 1) {
    const currentBit = (value >>> bit) & 1;
    if (!node.children[currentBit]) {
      node.children[currentBit] = createTrieNode();
    }
    node = node.children[currentBit];
  }
}

function queryMaximumXor(root, value) {
  let node = root;
  let result = 0;

  for (let bit = 31; bit >= 0; bit -= 1) {
    const currentBit = (value >>> bit) & 1;
    const preferredBit = currentBit ^ 1;

    if (node.children[preferredBit]) {
      result = (result | ((1 << bit) >>> 0)) >>> 0;
      node = node.children[preferredBit];
    } else {
      node = node.children[currentBit];
    }
  }

  return result >>> 0;
}

function maximumSubarrayXor(values) {
  if (!Array.isArray(values) || values.length === 0) {
    throw new RangeError('values must be a non-empty array');
  }

  if (!values.every((value) => Number.isInteger(value) && value >= 0 && value <= 0xffffffff)) {
    throw new RangeError('values must contain unsigned 32-bit integers');
  }

  const root = createTrieNode();
  insertTrieValue(root, 0);

  let prefixXor = 0;
  let best = 0;

  for (const value of values) {
    prefixXor = (prefixXor ^ (value >>> 0)) >>> 0;
    best = Math.max(best, queryMaximumXor(root, prefixXor));
    insertTrieValue(root, prefixXor);
  }

  return best >>> 0;
}
```

## Common Pitfalls

- Recomputing every subarray XOR from scratch instead of using prefix XORs.
- Forgetting to insert prefix `0`, which misses subarrays that start at index `0`.
- Treating negative signed bitwise results as if they were already normalized unsigned values.
