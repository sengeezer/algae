## Why It Surfaces

Booth's algorithm is less about raw multiplication than about understanding signed binary arithmetic at register level. The key insight is that a run of multiplier ones can be handled by one subtraction when the run starts and one addition when it ends, with arithmetic shifts propagating the sign.

## Worked Example

Multiplying `0110` (`6`) by `0010` (`2`) with 4-bit operands yields `00001100`, which is the 8-bit two's-complement representation of `12`.

  if (typeof value !== 'string' || !/^[01]+$/.test(value)) {

function toSignedBigInt(binary) {
  const width = BigInt(binary.length);
  const unsigned = BigInt(`0b${binary}`);
  const signBit = 1n << (width - 1n);

  return (unsigned & signBit) === 0n ? unsigned : unsigned - (1n << width);
}

function toUnsignedBigInt(value, width) {
  return value & ((1n << BigInt(width)) - 1n);
}

function wrapToWidth(value, width) {
  const unsigned = toUnsignedBigInt(value, width);
  const signBit = 1n << BigInt(width - 1);

  return (unsigned & signBit) === 0n ? unsigned : unsigned - (1n << BigInt(width));
}

function boothsMultiplicationAlgorithm(multiplicand, multiplier) {
  validateSignedBinaryOperand(multiplicand, 'multiplicand');
  validateSignedBinaryOperand(multiplier, 'multiplier');

  if (multiplicand.length !== multiplier.length) {
    throw new RangeError('Operands must use the same bit width');
  }

  const bitWidth = multiplicand.length;
  const registerMask = (1n << BigInt(bitWidth)) - 1n;
  const br = toSignedBigInt(multiplicand);

  let ac = 0n;
  let qr = BigInt(`0b${multiplier}`);
  let qMinusOne = 0n;

  for (let step = 0; step < bitWidth; step += 1) {
    const qrLeastSignificantBit = qr & 1n;

    if (qrLeastSignificantBit === 0n && qMinusOne === 1n) {
      ac = wrapToWidth(ac + br, bitWidth);
    } else if (qrLeastSignificantBit === 1n && qMinusOne === 0n) {
      ac = wrapToWidth(ac - br, bitWidth);
    }

    const nextQMinusOne = qrLeastSignificantBit;
    const acLeastSignificantBit = toUnsignedBigInt(ac, bitWidth) & 1n;

    qr = ((qr >> 1n) | (acLeastSignificantBit << BigInt(bitWidth - 1))) & registerMask;
    ac = wrapToWidth(ac >> 1n, bitWidth);
    qMinusOne = nextQMinusOne;
  }

  const product = (toUnsignedBigInt(ac, bitWidth) << BigInt(bitWidth)) | qr;
  return product.toString(2).padStart(bitWidth * 2, '0');
}

## Common Pitfalls

- Treating the operands as unsigned even though Booth's algorithm is designed for signed two's-complement values.
- Using logical right shift semantics where the accumulator requires arithmetic sign extension.
- Forgetting to constrain accumulator updates back to the chosen bit width after each add or subtract.
