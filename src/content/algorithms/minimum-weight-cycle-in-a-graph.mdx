## Why It Surfaces

The problem looks like generic cycle analysis, but the weighted objective changes the right approach. Instead of enumerating cycles directly, treat every edge as a candidate closing edge and ask for the shortest way to reconnect its two endpoints without using that edge.

## Worked Example

If edge `(0, 4)` has weight `3`, and the cheapest remaining path from `0` to `4` is `0 -> 1 -> 4` with total cost `3`, then that edge closes a cycle of weight `6`. Repeating that check for every edge reveals the minimum-weight cycle.

```js
function minimumWeightCycleInAGraph(nodeCount, edges) {
  const graph = Array.from({ length: nodeCount }, () => []);
  edges.forEach(([from, to, weight], index) => {
    graph[from].push([to, weight, index]);
    graph[to].push([from, weight, index]);
  });

  function shortestPath(source, destination, blockedEdgeIndex) {
    const distance = new Array(nodeCount).fill(Number.POSITIVE_INFINITY);
    const queue = [[0, source]];
    distance[source] = 0;

    while (queue.length > 0) {
      queue.sort((left, right) => left[0] - right[0]);
      const [cost, node] = queue.shift();
      if (cost > distance[node]) {
        continue;
      }

      for (const [neighbor, weight, edgeIndex] of graph[node]) {
        if (edgeIndex === blockedEdgeIndex) {
          continue;
        }

        const candidate = cost + weight;
        if (candidate < distance[neighbor]) {
          distance[neighbor] = candidate;
          queue.push([candidate, neighbor]);
        }
      }
    }

    return distance[destination];
  }

  let bestCycle = Number.POSITIVE_INFINITY;
  edges.forEach(([from, to, weight], edgeIndex) => {
    const pathCost = shortestPath(from, to, edgeIndex);
    if (pathCost !== Number.POSITIVE_INFINITY) {
      bestCycle = Math.min(bestCycle, pathCost + weight);
    }
  });

  return bestCycle === Number.POSITIVE_INFINITY ? -1 : bestCycle;
}
```

  nodeCount: number,
  edges: Array<[number, number, number]>,
): number {
  const graph = Array.from({ length: nodeCount }, () => [] as Array<[number, number, number]>);
  edges.forEach(([from, to, weight], index) => {
    graph[from].push([to, weight, index]);
    graph[to].push([from, weight, index]);
  });

  function shortestPath(source: number, destination: number, blockedEdgeIndex: number): number {
    const distance = new Array<number>(nodeCount).fill(Number.POSITIVE_INFINITY);
    const queue: Array<[number, number]> = [[0, source]];
    distance[source] = 0;

    while (queue.length > 0) {
      queue.sort((left, right) => left[0] - right[0]);
      const [cost, node] = queue.shift()!;
      if (cost > distance[node]) {
        continue;
      }

      for (const [neighbor, weight, edgeIndex] of graph[node]) {
        if (edgeIndex === blockedEdgeIndex) {
          continue;
        }

        const candidate = cost + weight;
        if (candidate < distance[neighbor]) {
          distance[neighbor] = candidate;
          queue.push([candidate, neighbor]);
        }
      }
    }

    return distance[destination];
  }

  let bestCycle = Number.POSITIVE_INFINITY;
  edges.forEach(([from, to, weight], edgeIndex) => {
    const pathCost = shortestPath(from, to, edgeIndex);
    if (pathCost !== Number.POSITIVE_INFINITY) {
      bestCycle = Math.min(bestCycle, pathCost + weight);
    }
  });

  return bestCycle === Number.POSITIVE_INFINITY ? -1 : bestCycle;
}

## Common Pitfalls

- Looking only for any cycle instead of the lightest one.
- Forgetting to exclude the candidate edge while computing the reconnecting path.
- Assuming the locally lightest edge must belong to the globally lightest cycle.
