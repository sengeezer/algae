## Why It Surfaces

This is the rectangle version of the earlier point-in-triangle predicate. Once the rectangle vertices are given in perimeter order, the query point is inside exactly when the four triangle areas created with the edges add up to the area of the rectangle itself.

## Worked Example

For rectangle `[[10, 10], [10, -10], [-10, -10], [-10, 10]]` and point `[0, 0]`, the triangle-area sum matches the rectangle area, so the point lies inside.

## Reference Implementations

<ExamplePair>

```js
function checkWhetherAGivenPointLiesInsideARectangleOrNot(rectangle, point) {
  if (rectangle.length !== 4) {
    throw new RangeError("Rectangle must contain exactly four vertices in perimeter order");
  }

  function polygonArea(points) {
    let total = 0;

    for (let index = 0; index < points.length; index += 1) {
      const next = (index + 1) % points.length;
      total += points[index][0] * points[next][1] - points[next][0] * points[index][1];
    }

    return Math.abs(total) / 2;
  }

  function triangleArea(a, b, c) {
    return Math.abs(
      a[0] * (b[1] - c[1])
      + b[0] * (c[1] - a[1])
      + c[0] * (a[1] - b[1])
    ) / 2;
  }

  const whole = polygonArea(rectangle);

  if (whole === 0) {
    return false;
  }

  let parts = 0;

  for (let index = 0; index < rectangle.length; index += 1) {
    const next = (index + 1) % rectangle.length;
    parts += triangleArea(rectangle[index], rectangle[next], point);
  }

  return Math.abs(whole - parts) < 1e-9;
}
```

export function checkWhetherAGivenPointLiesInsideARectangleOrNot(
  rectangle: Point[],
  point: Point,
): boolean {
  if (rectangle.length !== 4) {
    throw new RangeError('Rectangle must contain exactly four vertices in perimeter order');
  }

  const polygonArea = (points: Point[]): number => {
    let total = 0;

    for (let index = 0; index < points.length; index += 1) {
      const next = (index + 1) % points.length;
      total += points[index][0] * points[next][1] - points[next][0] * points[index][1];
    }

    return Math.abs(total) / 2;
  };

  const triangleArea = (a: Point, b: Point, c: Point): number => {
    return Math.abs(
      a[0] * (b[1] - c[1])
      + b[0] * (c[1] - a[1])
      + c[0] * (a[1] - b[1])
    ) / 2;
  };

  const whole = polygonArea(rectangle);

  if (whole === 0) {
    return false;
  }

  let parts = 0;

  for (let index = 0; index < rectangle.length; index += 1) {
    const next = (index + 1) % rectangle.length;
    parts += triangleArea(rectangle[index], rectangle[next], point);
  }

  return Math.abs(whole - parts) < 1e-9;
}

## Follow-up Questions

- How would you adapt this if the rectangle were guaranteed axis-aligned?
- Why do the vertices need to be supplied in perimeter order for this exact area decomposition?
- How would you exclude boundary points from the predicate?

## Common Pitfalls

- Assuming the rectangle is axis-aligned when the prompt allows rotation.
- Treating an arbitrary permutation of the four vertices as if it were polygon order.
- Using exact floating-point equality instead of a tolerance.
