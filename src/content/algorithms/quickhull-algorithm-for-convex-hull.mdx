## Why It Surfaces

Quickhull is useful when you want an interview-friendly divide-and-conquer hull without implementing a full tangent merge. The farthest point from a segment must lie on the hull, and every point inside the triangle formed by that point and the segment endpoints can be thrown away immediately.

## Worked Example

For `[[0, 3], [1, 1], [2, 2], [4, 4], [0, 0], [1, 2], [3, 1], [3, 3]]`, the outer hull is `[[0, 0], [3, 1], [4, 4], [0, 3]]` in counterclockwise order.

function quickhullAlgorithmForConvexHull(points) {
  function uniqueSorted(items) {
    return [...new Set(items.map(([x, y]) => `${x},${y}`))]
      .map((key) => key.split(",").map(Number))
      .sort((left, right) => (left[0] - right[0]) || (left[1] - right[1]));
  }

  function cross(a, b, c) {
    return (b[0] - a[0]) * (c[1] - a[1]) - (b[1] - a[1]) * (c[0] - a[0]);
  }

  function monotonicChain(items) {
    if (items.length <= 1) {
      return items.map((point) => [...point]);
    }

    const sorted = uniqueSorted(items);
    const lower = [];

    for (const point of sorted) {
      while (lower.length >= 2 && cross(lower[lower.length - 2], lower[lower.length - 1], point) <= 0) {
        lower.pop();
      }

      lower.push(point);
    }

    const upper = [];

    for (let index = sorted.length - 1; index >= 0; index -= 1) {
      const point = sorted[index];

      while (upper.length >= 2 && cross(upper[upper.length - 2], upper[upper.length - 1], point) <= 0) {
        upper.pop();
      }

      upper.push(point);
    }

    lower.pop();
    upper.pop();
    return [...lower, ...upper].map((point) => [...point]);
  }

  const uniquePoints = uniqueSorted(points);

  if (uniquePoints.length <= 2) {
    return uniquePoints;
  }

  const leftMost = uniquePoints[0];
  const rightMost = uniquePoints[uniquePoints.length - 1];
  const hullPoints = new Set([`${leftMost[0]},${leftMost[1]}`, `${rightMost[0]},${rightMost[1]}`]);

  function addHull(candidates, start, end, side) {
    let farthestPoint = null;
    let maxDistance = 0;

    for (const point of candidates) {
      const signedArea = cross(start, end, point);

      if ((side === 1 && signedArea <= 0) || (side === -1 && signedArea >= 0)) {
        continue;
      }

      const distance = Math.abs(signedArea);

      if (distance > maxDistance) {
        maxDistance = distance;
        farthestPoint = point;
      }
    }

    if (farthestPoint === null) {
      return;
    }

    hullPoints.add(`${farthestPoint[0]},${farthestPoint[1]}`);
    addHull(candidates, start, farthestPoint, -Math.sign(cross(start, farthestPoint, end) || side));
    addHull(candidates, farthestPoint, end, -Math.sign(cross(farthestPoint, end, start) || side));
  }

  addHull(uniquePoints, leftMost, rightMost, 1);
  addHull(uniquePoints, leftMost, rightMost, -1);

  return monotonicChain(
    [...hullPoints].map((key) => key.split(",").map(Number)),
  );
}

## Common Pitfalls

- Forgetting that at least three distinct points are needed for a non-degenerate hull.
- Returning hull points in arbitrary set order instead of an actual boundary walk.
- Failing to de-duplicate repeated points before the recursion starts.
