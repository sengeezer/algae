## Why It Surfaces

This is what happens when the usual LIS recurrence becomes a counting problem. For each value, you want the sum of counts for all smaller values seen so far, so the bottleneck is no longer choosing a transition but aggregating many transitions quickly.

## Worked Example

For `[1, 2, 3]`, every non-empty subsequence is increasing, so the answer is `7`. For `[3, 2, 1]`, only the three single-element subsequences qualify, so the answer is `3`.

class BigIntSumSegmentTree {
  constructor(size) {
    this.size = size;
    this.tree = new Array(size * 4).fill(0n);
  }

  add(index, delta) {
    this.addNode(1, 0, this.size - 1, index, delta);
  }

  addNode(node, left, right, index, delta) {
    if (left === right) {
      this.tree[node] += delta;
      return;
    }

    const mid = Math.floor((left + right) / 2);
    if (index <= mid) {
      this.addNode(node * 2, left, mid, index, delta);
    } else {
      this.addNode(node * 2 + 1, mid + 1, right, index, delta);
    }

    this.tree[node] = this.tree[node * 2] + this.tree[node * 2 + 1];
  }

  query(queryLeft, queryRight) {
    if (queryLeft > queryRight) {
      return 0n;
    }

    return this.queryNode(1, 0, this.size - 1, queryLeft, queryRight);
  }

  queryNode(node, left, right, queryLeft, queryRight) {
    if (queryRight < left || right < queryLeft) {
      return 0n;
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return (
      this.queryNode(node * 2, left, mid, queryLeft, queryRight)
      + this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight)
    );
  }
}

function countNumberOfIncreasingSubSequences(values) {
  if (!Array.isArray(values) || values.length === 0) {
    return 0n;
  }

  const ranks = [...new Set(values)].sort((left, right) => left - right);
  const rankByValue = new Map(ranks.map((value, index) => [value, index]));
  const tree = new BigIntSumSegmentTree(ranks.length);

  let total = 0n;

  for (const value of values) {
    const rank = rankByValue.get(value);
    const extendableCount = tree.query(0, rank - 1);
    const currentCount = 1n + extendableCount;
    tree.add(rank, currentCount);
    total += currentCount;
  }

  return total;
}

class BigIntSumSegmentTree {
  private readonly size: number;
  private readonly tree: bigint[];

  constructor(size: number) {
    this.size = size;
    this.tree = new Array<bigint>(size * 4).fill(0n);
  }

  add(index: number, delta: bigint): void {
    this.addNode(1, 0, this.size - 1, index, delta);
  }

  query(queryLeft: number, queryRight: number): bigint {
    if (queryLeft > queryRight) {
      return 0n;
    }

    return this.queryNode(1, 0, this.size - 1, queryLeft, queryRight);
  }

  private addNode(node: number, left: number, right: number, index: number, delta: bigint): void {
    if (left === right) {
      this.tree[node] += delta;
      return;
    }

    const mid = Math.floor((left + right) / 2);
    if (index <= mid) {
      this.addNode(node * 2, left, mid, index, delta);
    } else {
      this.addNode(node * 2 + 1, mid + 1, right, index, delta);
    }

    this.tree[node] = this.tree[node * 2] + this.tree[node * 2 + 1];
  }

  private queryNode(
    node: number,
    left: number,
    right: number,
    queryLeft: number,
    queryRight: number,
  ): bigint {
    if (queryRight < left || right < queryLeft) {
      return 0n;
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return (
      this.queryNode(node * 2, left, mid, queryLeft, queryRight)
      + this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight)
    );
  }
}

export function countNumberOfIncreasingSubSequences(values: number[]): bigint {
  if (values.length === 0) {
    return 0n;
  }

  const ranks = [...new Set(values)].sort((left, right) => left - right);
  const rankByValue = new Map<number, number>(ranks.map((value, index) => [value, index]));
  const tree = new BigIntSumSegmentTree(ranks.length);

  let total = 0n;

  for (const value of values) {
    const rank = rankByValue.get(value);
    if (rank === undefined) {
      throw new Error('Missing compressed rank');
    }

    const extendableCount = tree.query(0, rank - 1);
    const currentCount = 1n + extendableCount;
    tree.add(rank, currentCount);
    total += currentCount;
  }

  return total;
}

## Follow-up Questions

- How would you add a modulus if the exact count becomes too large?
- Why must duplicates query only strictly smaller ranks?
- Could a Fenwick tree replace the segment tree here without changing the recurrence?

## Common Pitfalls

- Counting non-decreasing subsequences by accident when duplicates are present.
- Forgetting to coordinate-compress before building the tree.
- Using `number` for counts and silently overflowing on large inputs.
