## Why It Surfaces

This problem looks like plain string splitting until you notice that numeric overflow is part of the trap. The safe approach keeps everything as strings, adds digit by digit, and checks whether the next substring matches the expected sum.

## Worked Example

`12243660` is a sum-string because it can be split as `12, 24, 36, 60`, and each number equals the sum of the previous two. `123456` fails because no opening split makes the rest of the string line up with repeated sums.

function isSumString(text) {
  if (typeof text !== 'string' || /[^0-9]/.test(text)) {
    throw new TypeError('Input must be a digit string');
  }

  for (let firstLength = 1; firstLength <= Math.floor(text.length / 2); firstLength += 1) {
    for (let secondLength = 1; firstLength + secondLength < text.length; secondLength += 1) {
      if (hasLeadingZero(text.slice(0, firstLength)) || hasLeadingZero(text.slice(firstLength, firstLength + secondLength))) {
        continue;
      }

      if (checkSequence(text, 0, firstLength, secondLength)) {
        return true;
      }
    }
  }

  return false;
}

function checkSequence(text, start, firstLength, secondLength) {
  const first = text.slice(start, start + firstLength);
  const second = text.slice(start + firstLength, start + firstLength + secondLength);
  const sum = addStrings(first, second);
  const nextStart = start + firstLength + secondLength;

  if (!text.startsWith(sum, nextStart)) {
    return false;
  }

  if (nextStart + sum.length === text.length) {
    return true;
  }

  return checkSequence(text, start + firstLength, secondLength, sum.length);
}

function addStrings(left, right) {
  let carry = 0;
  let result = '';
  let leftIndex = left.length - 1;
  let rightIndex = right.length - 1;

  while (leftIndex >= 0 || rightIndex >= 0 || carry > 0) {
    const digitLeft = leftIndex >= 0 ? Number(left[leftIndex]) : 0;
    const digitRight = rightIndex >= 0 ? Number(right[rightIndex]) : 0;
    const total = digitLeft + digitRight + carry;
    result = String(total % 10) + result;
    carry = Math.floor(total / 10);
    leftIndex -= 1;
    rightIndex -= 1;
  }

  return result;
}

function hasLeadingZero(text) {
  return text.length > 1 && text[0] === '0';
}

## Common Pitfalls

- Allowing segments like `01` or `001` even though leading zeroes are forbidden.
- Parsing long substrings into native numbers and overflowing.
- Forgetting that the string must split into more than two numbers.
