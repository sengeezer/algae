## Why It Surfaces

The source article is really a small family of pyramid-volume formulas. The clean local abstraction is to treat the problem as a base-shape dispatch where each case computes base area and then applies the `1 / 3` pyramid-volume factor.

## Worked Example

For a square pyramid with base side `4` and height `9`, the volume is `(1 / 3) * 4^2 * 9 = 48`.

function programForVolumeOfPyramid(pyramid) {
  if (!pyramid || typeof pyramid !== "object" || typeof pyramid.baseType !== "string") {
    throw new RangeError("Pyramid description must include a baseType");
  }

  switch (pyramid.baseType) {
    case "square": {
      const { side, height } = pyramid;
      if (![side, height].every(Number.isFinite) || side < 0 || height < 0) {
        throw new RangeError("Square pyramid dimensions must be non-negative numbers");
      }

      return (side * side * height) / 3;
    }
    case "triangular": {
      const { base, altitude, height } = pyramid;
      if (![base, altitude, height].every(Number.isFinite) || base < 0 || altitude < 0 || height < 0) {
        throw new RangeError("Triangular pyramid dimensions must be non-negative numbers");
      }

      return (base * altitude * height) / 6;
    }
    case "pentagonal": {
      const { side, apothem, height } = pyramid;
      if (![side, apothem, height].every(Number.isFinite) || side < 0 || apothem < 0 || height < 0) {
        throw new RangeError("Pentagonal pyramid dimensions must be non-negative numbers");
      }

      return (5 * side * apothem * height) / 6;
    }
    case "hexagonal": {
      const { side, apothem, height } = pyramid;
      if (![side, apothem, height].every(Number.isFinite) || side < 0 || apothem < 0 || height < 0) {
        throw new RangeError("Hexagonal pyramid dimensions must be non-negative numbers");
      }

      return side * apothem * height;
    }
    default:
      throw new RangeError("Unsupported pyramid base type");
  }
}

## Follow-up Questions

## Common Pitfalls

- Using the source article's rounded constants instead of exact fractions.
- Mixing up the pyramid height with a base-side measurement.
- Treating all base shapes as if they used the same base-area formula.
