## Why It Surfaces

Kosaraju is the SCC algorithm that makes finishing order feel concrete. The first DFS does not return the components directly; it builds the order that makes the second DFS on the reversed graph reveal one SCC at a time.

## Worked Example

If `1 -> 2 -> 3 -> 1` forms one cycle and `4 -> 5 -> 6 -> 4` forms another, the first DFS records a finishing stack across the whole graph. Processing nodes from that stack on the reversed graph returns `[1, 2, 3]` and `[4, 5, 6]` as separate SCCs.

```js
function kosaraju(edges) {
  const nodes = new Set();
  const graph = new Map();
  const reversed = new Map();

  for (const [from, to] of edges) {
    nodes.add(from);
    nodes.add(to);
    if (!graph.has(from)) graph.set(from, []);
    if (!graph.has(to)) graph.set(to, []);
    if (!reversed.has(from)) reversed.set(from, []);
    if (!reversed.has(to)) reversed.set(to, []);
    graph.get(from).push(to);
    reversed.get(to).push(from);
  }

  const visited = new Set();
  const order = [];

  function dfsOrder(node) {
    visited.add(node);
    for (const neighbor of graph.get(node) ?? []) {
      if (!visited.has(neighbor)) {
        dfsOrder(neighbor);
      }
    }
    order.push(node);
  }

  for (const node of nodes) {
    if (!visited.has(node)) {
      dfsOrder(node);
    }
  }

  visited.clear();
  const components = [];

  function dfsCollect(node, component) {
    visited.add(node);
    component.push(node);
    for (const neighbor of reversed.get(node) ?? []) {
      if (!visited.has(neighbor)) {
        dfsCollect(neighbor, component);
      }
    }
  }

  while (order.length > 0) {
    const node = order.pop();
    if (!visited.has(node)) {
      const component = [];
      dfsCollect(node, component);
      components.push(component);
    }
  }

  return components;
}
```

```ts
export function kosaraju(edges: Array<[number, number]>): number[][] {
  const nodes = new Set<number>();
  const graph = new Map<number, number[]>();
  const reversed = new Map<number, number[]>();

  for (const [from, to] of edges) {
    nodes.add(from);
    nodes.add(to);
    if (!graph.has(from)) graph.set(from, []);
    if (!graph.has(to)) graph.set(to, []);
    if (!reversed.has(from)) reversed.set(from, []);
    if (!reversed.has(to)) reversed.set(to, []);
    graph.get(from)!.push(to);
    reversed.get(to)!.push(from);
  }

  const visited = new Set<number>();
  const order: number[] = [];

  function dfsOrder(node: number): void {
    visited.add(node);
    for (const neighbor of graph.get(node) ?? []) {
      if (!visited.has(neighbor)) {
        dfsOrder(neighbor);
      }
    }
    order.push(node);
  }

  for (const node of nodes) {
    if (!visited.has(node)) {
      dfsOrder(node);
    }
  }

  visited.clear();
  const components: number[][] = [];

  function dfsCollect(node: number, component: number[]): void {
    visited.add(node);
    component.push(node);
    for (const neighbor of reversed.get(node) ?? []) {
      if (!visited.has(neighbor)) {
        dfsCollect(neighbor, component);
      }
    }
  }

  while (order.length > 0) {
    const node = order.pop()!;
    if (!visited.has(node)) {
      const component: number[] = [];
      dfsCollect(node, component);
      components.push(component);
    }
  }

  return components;
}
```

## Common Pitfalls

- Forgetting to reverse the graph before the second pass.
- Running the second DFS in arbitrary order instead of reverse finishing order.
- Treating SCCs as if the graph were undirected connected components.
