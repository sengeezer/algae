## Why It Surfaces

This problem is a good test of whether you can invent the right node summary for a segment tree. Instead of storing one number, each interval keeps enough information to say how many pairs are already matched and how many opens or closes remain available to match across a boundary.

## Worked Example

For `"())(())(())("`, the query `[0, 11]` returns `10` because the longest correct bracket subsequence has five matched pairs. The query `[1, 2]` returns `0` because `"))"` cannot form any valid pair.

function mergeBracketNodes(leftNode, rightNode) {
  const matchedAcrossBoundary = Math.min(leftNode.open, rightNode.closed);

  return {
    pairs: leftNode.pairs + rightNode.pairs + matchedAcrossBoundary,
    open: leftNode.open + rightNode.open - matchedAcrossBoundary,
    closed: leftNode.closed + rightNode.closed - matchedAcrossBoundary,
  };
}

class LongestCorrectBracketSubsequence {
  constructor(text) {
    if (typeof text !== 'string' || text.length === 0) {
      throw new RangeError('Text must be a non-empty string');
    }

    this.text = text;
    this.size = text.length;
    this.tree = new Array(this.size * 4);
    this.build(1, 0, this.size - 1);
  }

  build(node, left, right) {
    if (left === right) {
      this.tree[node] = {
        pairs: 0,
        open: this.text[left] === '(' ? 1 : 0,
        closed: this.text[left] === ')' ? 1 : 0,
      };
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = mergeBracketNodes(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  query(queryLeft, queryRight) {
    this.validateRange(queryLeft, queryRight);
    return this.queryNode(1, 0, this.size - 1, queryLeft, queryRight).pairs * 2;
  }

  queryNode(node, left, right, queryLeft, queryRight) {
    if (queryRight < left || right < queryLeft) {
      return { pairs: 0, open: 0, closed: 0 };
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return mergeBracketNodes(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }

  validateRange(queryLeft, queryRight) {
    if (
      !Number.isInteger(queryLeft)
      || !Number.isInteger(queryRight)
      || queryLeft < 0
      || queryRight < queryLeft
      || queryRight >= this.size
    ) {
      throw new RangeError('Query range is invalid');
    }
  }
}

  pairs: number;
  open: number;
  closed: number;
}

function mergeBracketNodes(leftNode: BracketNode, rightNode: BracketNode): BracketNode {
  const matchedAcrossBoundary = Math.min(leftNode.open, rightNode.closed);

  return {
    pairs: leftNode.pairs + rightNode.pairs + matchedAcrossBoundary,
    open: leftNode.open + rightNode.open - matchedAcrossBoundary,
    closed: leftNode.closed + rightNode.closed - matchedAcrossBoundary,
  };
}

  private build(node: number, left: number, right: number): void {
    if (left === right) {

## Common Pitfalls

- Treating this as a substring problem when the prompt asks for a subsequence.
- Forgetting that unmatched opens from the left can only pair with unmatched closes from the right.
- Returning the number of pairs instead of the length of the subsequence.
