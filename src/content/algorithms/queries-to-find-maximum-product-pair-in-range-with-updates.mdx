## Why It Surfaces

This is an interview-friendly example of a segment tree storing more than one summary value per node. Because all numbers are positive, the best product in a range comes from the two largest values, so every merge only needs to keep the top two candidates.

## Worked Example

For `[1, 3, 4, 2, 5]`, the query `[1, 0, 2]` returns `12` from `3 * 4`. After updating index `1` to `6`, the same range returns `24` from `6 * 4`.

function mergeTopTwo(leftNode, rightNode) {
  const values = [leftNode.largest, leftNode.secondLargest, rightNode.largest, rightNode.secondLargest]
    .filter((value) => value !== -Infinity)
    .sort((left, right) => right - left);

  return {
    largest: values[0] ?? -Infinity,
    secondLargest: values[1] ?? -Infinity,
  };
}

class MaximumProductPairSegmentTree {
  constructor(values) {
    if (!Array.isArray(values) || values.length < 2) {
      throw new RangeError('Values must contain at least two numbers');
    }

    if (values.some((value) => !Number.isFinite(value) || value <= 0)) {
      throw new RangeError('All values must be positive finite numbers');
    }

    this.values = [...values];
    this.size = values.length;
    this.tree = new Array(this.size * 4);
    this.build(1, 0, this.size - 1);
  }

  build(node, left, right) {
    if (left === right) {
      this.tree[node] = { largest: this.values[left], secondLargest: -Infinity };
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = mergeTopTwo(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  update(index, value) {
    this.validateIndex(index);
    if (!Number.isFinite(value) || value <= 0) {
      throw new RangeError('Updated value must be a positive finite number');
    }

    this.values[index] = value;
    this.updateNode(1, 0, this.size - 1, index, value);
  }

  updateNode(node, left, right, index, value) {
    if (left === right) {
      this.tree[node] = { largest: value, secondLargest: -Infinity };
      return;
    }

    const mid = Math.floor((left + right) / 2);
    if (index <= mid) {
      this.updateNode(node * 2, left, mid, index, value);
    } else {
      this.updateNode(node * 2 + 1, mid + 1, right, index, value);
    }

    this.tree[node] = mergeTopTwo(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  query(queryLeft, queryRight) {
    this.validateRange(queryLeft, queryRight);
    const node = this.queryNode(1, 0, this.size - 1, queryLeft, queryRight);
    if (node.secondLargest === -Infinity) {
      throw new RangeError('Query range must contain at least two elements');
    }

    return node.largest * node.secondLargest;
  }

  queryNode(node, left, right, queryLeft, queryRight) {
    if (queryRight < left || right < queryLeft) {
      return { largest: -Infinity, secondLargest: -Infinity };
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return mergeTopTwo(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }

  validateIndex(index) {
    if (!Number.isInteger(index) || index < 0 || index >= this.size) {
      throw new RangeError('Index is out of bounds');
    }
  }

  validateRange(queryLeft, queryRight) {
    if (
      !Number.isInteger(queryLeft)
      || !Number.isInteger(queryRight)
      || queryLeft < 0
      || queryRight < queryLeft
      || queryRight >= this.size
    ) {
      throw new RangeError('Query range is invalid');
    }
  }
}

function queriesToFindMaximumProductPairInRangeWithUpdates(values, queries) {
  const tree = new MaximumProductPairSegmentTree(values);
  const results = [];

  for (const [type, left, rightOrValue] of queries) {
    if (type === 1) {
      results.push(tree.query(left, rightOrValue));
    } else if (type === 2) {
      tree.update(left, rightOrValue);
    } else {
      throw new RangeError(`Unknown query type: ${type}`);
    }
  }

  return results;
}

interface TopTwoNode {
  largest: number;
  secondLargest: number;
}

function mergeTopTwo(leftNode: TopTwoNode, rightNode: TopTwoNode): TopTwoNode {
  const values = [leftNode.largest, leftNode.secondLargest, rightNode.largest, rightNode.secondLargest]

    .sort((left, right) => right - left);

  return {
    largest: values[0] ?? -Infinity,
    secondLargest: values[1] ?? -Infinity,
  };
}

class MaximumProductPairSegmentTree {
  private readonly size: number;
  private readonly values: number[];
  private readonly tree: TopTwoNode[];

  constructor(values: number[]) {
    if (values.length < 2) {
      throw new RangeError('Values must contain at least two numbers');
    }

    if (values.some((value) => !Number.isFinite(value) || value <= 0)) {
      throw new RangeError('All values must be positive finite numbers');
    }

    this.size = values.length;
    this.values = [...values];
    this.tree = new Array<TopTwoNode>(this.size * 4);
    this.build(1, 0, this.size - 1);
  }

  update(index: number, value: number): void {
    this.validateIndex(index);
    if (!Number.isFinite(value) || value <= 0) {
      throw new RangeError('Updated value must be a positive finite number');
    }

    this.values[index] = value;
    this.updateNode(1, 0, this.size - 1, index, value);
  }

  query(queryLeft: number, queryRight: number): number {
    this.validateRange(queryLeft, queryRight);
    const node = this.queryNode(1, 0, this.size - 1, queryLeft, queryRight);
    if (node.secondLargest === -Infinity) {
      throw new RangeError('Query range must contain at least two elements');
    }

    return node.largest * node.secondLargest;
  }

  private build(node: number, left: number, right: number): void {
    if (left === right) {
      this.tree[node] = { largest: this.values[left], secondLargest: -Infinity };
      return;
    }

    const mid = Math.floor((left + right) / 2);
    this.build(node * 2, left, mid);
    this.build(node * 2 + 1, mid + 1, right);
    this.tree[node] = mergeTopTwo(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  private updateNode(node: number, left: number, right: number, index: number, value: number): void {
    if (left === right) {
      this.tree[node] = { largest: value, secondLargest: -Infinity };
      return;
    }

    const mid = Math.floor((left + right) / 2);
    if (index <= mid) {
      this.updateNode(node * 2, left, mid, index, value);
    } else {
      this.updateNode(node * 2 + 1, mid + 1, right, index, value);
    }

    this.tree[node] = mergeTopTwo(this.tree[node * 2], this.tree[node * 2 + 1]);
  }

  private queryNode(node: number, left: number, right: number, queryLeft: number, queryRight: number): TopTwoNode {
    if (queryRight < left || right < queryLeft) {
      return { largest: -Infinity, secondLargest: -Infinity };
    }

    if (queryLeft <= left && right <= queryRight) {
      return this.tree[node];
    }

    const mid = Math.floor((left + right) / 2);
    return mergeTopTwo(
      this.queryNode(node * 2, left, mid, queryLeft, queryRight),
      this.queryNode(node * 2 + 1, mid + 1, right, queryLeft, queryRight),
    );
  }

  private validateIndex(index: number): void {
    if (!Number.isInteger(index) || index < 0 || index >= this.size) {
      throw new RangeError('Index is out of bounds');
    }
  }

  private validateRange(queryLeft: number, queryRight: number): void {
    if (
      !Number.isInteger(queryLeft)
      || !Number.isInteger(queryRight)
      || queryLeft < 0
      || queryRight < queryLeft
      || queryRight >= this.size
    ) {
      throw new RangeError('Query range is invalid');
    }
  }
}

export function queriesToFindMaximumProductPairInRangeWithUpdates(
  values: number[],
  queries: Array<[1, number, number] | [2, number, number]>,
): number[] {
  const tree = new MaximumProductPairSegmentTree(values);
  const results: number[] = [];

  for (const [type, left, rightOrValue] of queries) {
    if (type === 1) {
      results.push(tree.query(left, rightOrValue));
    } else {
      tree.update(left, rightOrValue);
    }
  }

  return results;
}

## Common Pitfalls

- Forgetting that the article assumes all values are positive.
- Storing only the maximum value and losing the second factor needed for the answer.
- Returning the pair itself when the prompt asks only for the product.
