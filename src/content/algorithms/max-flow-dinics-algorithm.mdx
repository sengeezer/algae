## Why It Surfaces

Dinic's algorithm is a strong interview upgrade over plain augmenting-path max flow because it separates the work into two layers: BFS decides which edges are currently “forward enough,” and DFS pushes as much blocking flow as possible before the next BFS rebuild.

## Worked Example

After one BFS from source to sink, only edges that move from level `k` to `k + 1` remain usable. DFS then pushes flow through that level graph until every source-to-sink route in it is blocked, after which a new BFS rebuilds levels in the updated residual network.

```js
class Dinic {
  constructor(size) {
    this.size = size;
    this.graph = Array.from({ length: size }, () => []);
  }

  addEdge(from, to, capacity) {
    const forward = { to, rev: this.graph[to].length, capacity };
    const backward = { to: from, rev: this.graph[from].length, capacity: 0 };
    this.graph[from].push(forward);
    this.graph[to].push(backward);
  }

  buildLevels(source, sink) {
    this.level = new Array(this.size).fill(-1);
    const queue = [source];
    this.level[source] = 0;

    for (let head = 0; head < queue.length; head += 1) {
      const node = queue[head];

      for (const edge of this.graph[node]) {
        if (edge.capacity > 0 && this.level[edge.to] === -1) {
          this.level[edge.to] = this.level[node] + 1;
          queue.push(edge.to);
        }
      }
    }

    return this.level[sink] !== -1;
  }

  sendFlow(node, sink, pushed) {
    if (node === sink || pushed === 0) {
      return pushed;
    }

    for (; this.nextEdge[node] < this.graph[node].length; this.nextEdge[node] += 1) {
      const edge = this.graph[node][this.nextEdge[node]];
      if (edge.capacity === 0 || this.level[edge.to] !== this.level[node] + 1) {
        continue;
      }

      const flow = this.sendFlow(edge.to, sink, Math.min(pushed, edge.capacity));
      if (flow === 0) {
        continue;
      }

      edge.capacity -= flow;
      this.graph[edge.to][edge.rev].capacity += flow;
      return flow;
    }

    return 0;
  }

  maxFlow(source, sink) {
    let total = 0;

    while (this.buildLevels(source, sink)) {
      this.nextEdge = new Array(this.size).fill(0);
      let flow = this.sendFlow(source, sink, Number.POSITIVE_INFINITY);

      while (flow > 0) {
        total += flow;
        flow = this.sendFlow(source, sink, Number.POSITIVE_INFINITY);
      }
    }

    return total;
  }
}
```

## Common Pitfalls

- Breaking reverse-edge bookkeeping in the residual graph.
- Letting DFS ignore the BFS level ordering.
- Forgetting that capacities must be updated in both forward and reverse directions.
